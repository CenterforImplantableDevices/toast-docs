    '''Uses a network object to generate the standard training/testing loss plot.

[docs]def plot_distribution_weights( network, save_txt=True, show_plot=False, save_plot=True, save_path='./', output_folder_name='no_folder_name', path_log='./log.txt', close_plot=True):
    '''Uses a network object to generate distribution plot of the weights of each layer from the last training iteration

        This will make separate plots for different layer types, so that similar layer types can be easily compared.
        
        This is primarily a helper function to allow a single function call with a network object as the argument.
        For further plotting functionality description, the plotting specific actions are implemented by :py:func:`~toast.plate.heat_plot.layer_distributions`

        Note:
            **Linear Layers** contain a weights parameter module and a bias parameter module. (1 layer, 2 parameter modules)
            
            **Non-linearity Layers** do not contain parameters as they only perform a fixed operation.

        Args:
            network (toast network):            `toast.heat` generated network that has been trained so that it contains training history values to plot.
            save_txt (bool, optional):          Whether to save a .txt summary of the final state of the varaibles for each layer
            show_plot (bool, optional):         Whether to show the plot after generation
            save_plot (bool, optional):         Whether to save the plot after generation.  Note the filename will be determined based on attributes of the network's path_network attribute
            save_path (str, optional):          Path to the parent directory where output plots should be saved.
            output_folder_name (str, optional): Child directory to save the plots to.
            path_log (str, optional):           Path to the log file for logging activity.
            close_plot (bool, optional):        Whether to close the figure when finished. THis helps minimize memory usage.

        Returns:
            No return value. Generates and displays/saves a plot
    '''    
    import torch.nn as nn

    # Define a dictionary to hold and organize the layer types
    data_layers = { 'OtherLayers': [] }
    for key, layer in network.layer_type_dict.items():
        data_layers[key] = []

        if any([ (l == nn.Linear) for l in layer]):
            data_layers[key+'Bias'] = []
            data_layers[key+'Wght'] = []
            data_layers[key+'Unkn'] = []

    # Iterate through the layers with recorded activation values and organize them by layer type.
    for index, param in enumerate( network.parameters()):
        type_found = False

        # Iterate through all the expected layer types to see which type they match
        for layer_name, layer_types in network.layer_type_dict.items():
            if any( [isinstance( network.get_param_modules()[index], l_type) for l_type in layer_types]):
                # We found an expected layer type
                type_found = True
                # Note that Linear layers contain two parameter types, 2D weights parameter and 1D bias parameter.
                # These layers must be handled specially
                if isinstance( network.get_param_modules()[index], nn.Linear):
                    if (len(param.shape) == 2):
                        data_layers[layer_name+'Wght'].append( param.detach().cpu().numpy().reshape((-1,)))
                    elif (len(param.shape) == 1):
                        data_layers[layer_name+'Bias'].append(  param.detach().cpu().numpy().reshape((-1,)))
                    else:
                        print('\n\nWARNING: UNEXPECTED PARAMETER OF STRANGE DIMENSION ATTATCHED TO LINEAR LAYER\n\n')
                        data_layers[layer_name+'Unkn'].append(  param.detach().cpu().numpy().reshape((-1,)))
                    break
                # All other layers get added normally
                else:
                    data_layers[layer_name].append( param.detach().cpu().numpy().reshape((-1,)))
                break
        # If an expected layer type is not found, add it to OtherLayers
        if not type_found:
            print('\nALERT: FOUND PARAMETER ATTACHED TO UNEXPECTEC MODULE: ', type( network.get_param_modules()[index]), '\n')
            data_layers['OtherLayers'].append(  param.detach().cpu().numpy().reshape((-1,)))

    if not os.path.exists( save_path):
        os.mkdir( save_path)
    # Plot the data included in each list sorted above, organized by layer type.
    for layer_name, data in data_layers.items():
        if len(data) > 1:
            try:
                num_bins = min( data[-2].shape[-1] * 2, 100)
            except:
                num_bins = None

            data_desc = 'layerDistWgt-' + layer_name # WT for Weights
            heat_plot.layer_distributions( data, bins=num_bins, save_txt=save_txt, title=network.path_network.split('/')[-1]+' '+data_desc, x_label='Parameter Distribution', show_plot=show_plot, save_plot=save_plot, save_path=save_path, output_folder_name=output_folder_name, description=network.path_network.split('/')[-1]+'_'+data_desc, path_log=path_log, bkg_color='antiquewhite', close_plot=close_plot)

    '''Uses a network object to generate distribution plot of the outputs produced from a dataset, organized by true label and combined to show full combined output spread

        This is primarily a helper function to allow a single function call with a network object as the argument.
        For further plotting functionality description, the plotting specific actions are implemented by :py:func:`~toast.plate.heat_plot.layer_distributions`

        Args:
            network (toast network):            `toast.heat` generated network that has been trained so that it contains training history values to plot.
            dataloader ( toast dataloader):     Dataloader object that can load the labels and samples to produce the distribution of outputs
            save_txt (bool, optional):          Whether to save a .txt summary of the final state of the varaibles for each layer
            show_plot (bool, optional):         Whether to show the plot after generation
            save_plot (bool, optional):         Whether to save the plot after generation.  Note the filename will be determined based on attributes of the network's path_network attribute
            save_path (str, optional):          Path to the parent directory where output plots should be saved.
            output_folder_name (str, optional): Child directory to save the plots to.
            path_log (str, optional):           Path to the log file for logging activity.
            close_plot (bool, optional):            Whether to close the figure when finished. THis helps minimize memory usage.

        Returns:
            No return value. Generate and displays/saves a plot
    '''
    labels, outputs = network.get_labeled_outputs_from( dataloader)

    if network.is_regenerator:
        print('Output Distribution plots for regenerator networks is not yet supported. Skipping plotting', end='\r')
        return
    
    output_distribution = []
    output_labels = []
    # Get all the unique labels. Sort them if possible
    try:
        unique_labels = sorted( unique( labels, axis=0))
    except:
        unique_labels = unique( labels, axis=0)
    
    # Organize the outputs based on what unique label they are associated with, creating 2 matching lists
    for unique_label in unique_labels:
        indicies = where(( labels == unique_label).all(axis=1))
        output_distribution.append( outputs[indicies])
        output_labels.append( unique_label)
    # Also combine all the outputs to create a cumulative distribution as an additional "output"
    output_distribution.append( outputs)
    output_labels.append('All Labels')

    heat_plot.layer_distributions( output_distribution, bins=100, save_txt=save_txt, title=network.path_network.split('/')[-1], x_label='Output Distribution', y_label=output_labels, show_plot=show_plot, save_plot=save_plot, save_path=save_path, output_folder_name=output_folder_name, description=network.path_network.split('/')[-1]+'_outputDist', path_log=path_log, bkg_color='antiquewhite', close_plot=close_plot)

    '''Uses a network object to generate plots comparing the input waveform to the output waveform. Used with regeneration networks to visualize their encoding/decoding accuracy.

        Different examples from different percentiles in the range of observed loss are generated, based on user inputs.

        This is primarily a helper function to allow a single function call with a network object as the argument.
        For further plotting functionality description, the plotting specific actions are implemented by :py:func:`~toast.plate.heat_plot.input_output_compare`

        Args:
            network (toast network):               `toast.heat` generated network that has been trained so that it can regenerate an input
            dataloader (toast dataloader):         Dataloader object that can load the labels and samples to produce the distribution of outputs
            loss_fn (torch.nn loss fcn, optional): Torch.nn derived loss function to use when calculating loss
            percentile_show (list, optional):      A list of which percentiles of loss to generate an example for.
            show_plot (bool, optional):            Whether to show the plot after generation
            save_plot (bool, optional):            Whether to save the plot after generation.  Note the filename will be determined based on attributes of the network's path_network attribute
            save_path (str, optional):             Path to the parent directory where output plots should be saved.
            output_folder_name (str, optional):    Child directory to save the plots to.
            path_log (str, optional):              Path to the log file for logging activity.
            close_plot (bool, optional):           Whether to close the figure when finished. THis helps minimize memory usage.

        Returns:
            No return value. Generate and displays/saves a plot
    '''
    if not network.is_regenerator:
        print('Cannot plot regenerator output for non-regenerator networks. Skipping plotting for ', network.path_network.split('/')[-1]+'_inputOutputComp', end='\r')
        return

    data_shape  = next( iter( dataloader))[0].shape[1:]
    num_batches = len(dataloader)
    samples_eachBatch = empty(( len(percentile_show), num_batches, *data_shape))
    outputs_eachBatch = empty(( len(percentile_show), num_batches, *data_shape))
    losses_eachBatch  = empty(( len(percentile_show), num_batches))

    # Set network to eval mode, which will turn off dropout layers, stop tracking the gradients/computation graphs, etc for faster performance
    network.eval()
    loss_fn.reduction = 'none'
    with torch.no_grad():
        # Test one full iteration of the samples provided
        for index_batch, (samples, _) in enumerate( dataloader):
            samples = samples.to(network.device)
            outputs = network.forward( samples)
            loss    = loss_fn(outputs, samples)

            while len(loss.shape) > 1:
                loss = loss.sum(axis=-1)

            # Get index of the sample closest to the selected percentile
            index_sorted = [i for _,i in sorted( zip( loss, range(len(loss))))]

            index_select = [ int(p_show * samples.shape[0]) for p_show in percentile_show]
            for j, select in enumerate(index_select):
                desired_sample = index_sorted[select]
                samples_eachBatch[j, index_batch, :] = samples[ desired_sample, :].cpu().detach().squeeze().numpy()
                outputs_eachBatch[j, index_batch, :] = outputs[ desired_sample, :].cpu().detach().squeeze().numpy()
                losses_eachBatch[ j, index_batch]    = loss[desired_sample]

    samples_final = empty(( len(percentile_show), *data_shape))
    outputs_final = empty(( len(percentile_show), *data_shape))
    for j, select in enumerate(index_select):
        index_sorted   = [i for _,i in sorted( zip( losses_eachBatch[j,:], range(num_batches)))]
        desired_sample = index_sorted[ num_batches // 2 ]
        samples_final[j,:] = samples_eachBatch[j, desired_sample, :]
        outputs_final[j,:] = outputs_eachBatch[j, desired_sample, :]

    titles = [ network.path_network.split('/')[-1] + '\nPercentile ' + str(p) for p in percentile_show]
    heat_plot.input_output_compare( samples_final, outputs_final, title=titles, separate_channels=separate_channels, show_plot=show_plot, save_plot=save_plot, save_path=save_path, output_folder_name=output_folder_name, description=network.path_network.split('/')[-1]+'_inputOutputComp', path_log=path_log, bkg_color='antiquewhite', close_plot=close_plot)

    '''Tests a network object and a Data_Bread class instance to assure proper function.

        Returns:
            No return value. Should execute without errors if successful.
    '''
    from sys import exit

    tlog.output_log('Network has ' + str(sum([torch.numel(p) for p in network.parameters()])) + ' parameters.', path_log)

    # print(next(iter(dataset.training)))
    test_sample, test_label = next(iter(dataset.training))
    # Add a batch dimension to test_sample
    test_sample = test_sample.unsqueeze(0)
    # test_sample = torch.from_numpy(test_sample).float()
    tlog.output_log('Test Sample: ' + str(test_sample) + ',   size: '+str(test_sample.size()), path_log)
    tlog.output_log('Test Label:  ' + str(test_label) + ',   size: '+str(test_label.size()), path_log)
    tlog.output_log('Sample Type: ' + str( type( test_sample)), path_log)
    tlog.output_log('Label  Type:  ' + str( type( test_label)), path_log)
    time_start = time.time()
    test_output = network( test_sample)
    tlog.output_log('Output: ' + str( test_output) + ',   size: '+str(test_output.size()), path_log)
    tlog.output_log('Forward Time: ' + str(time.time() - time_start), path_log)
    tlog.output_log('Output Type:  ' + str(type( test_output)), path_log)
    tlog.output_log('Output Data:  ' + str(test_output.data), path_log)

    if check_dataset:
        print('Checking Dataset...')
        if dataset.has_validation:
            for train_sample, _ in dataset.training:
              for test_sample, _ in dataset.testing:
                if torch.equal(test_sample, train_sample):
                  print('DATASET HAS REPEAT SAMPLES IN TRAINING AND TESTING')
                  exit()
                for val_sample, _ in dataset.validation:
                  if torch.equal(val_sample, train_sample):
                    print('DATASET HAS REPEAT SAMPLES IN TRAINING AND VALIDATION')
                    exit()
                  if torch.equal(val_sample, test_sample):
                    print('DATASET HAS REPEAT SAMPLES IN TESTING AND VALIDATION')
                    exit()
        else:
