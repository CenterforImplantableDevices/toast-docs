[docs]def plot_labels( all_labels_dataframe, save_plots, save_path, filename_header, col_x_axis='datetime', label_key={2:'before',3:'after',4:'between'}, plot_skip=[], save_pkl=False, plot_size=[16,9], plot_dpi=100, plot_linewidth=0.5, plot_close=True):
    '''General funcitons used for plotting output to check the resulting labeling data.

    Args:
        all_labels_dataframe (dataframe): Dataframe containing all raw data
        save_plots (bool):                Whether to save the plots
        save_path (str):                  Path to save the data (if save_plots is true)
        filename_header (str):            Prefix to use for filename to identify the contents of the plot
        col_x_axis (str, optional):       Column in all_labels_dataframe to use for the x axis.
        label_key (dict, optional):       Key for the label_ints and the string label. Used in x axis label.
        plot_skip (list, optional):       List of strings indicating a plot to skip. Options include `session`, `event_startStop`, `time_beforeEvent`, `time_afterEvent`, `time_duringEvent`, `labels`, `label_int`
        save_pkl (bool, optional):        Whether to save figure in a pickle file.  Note these files can be large files (>1GB).
        plot_size (list, optional):       Dimensions of each output figure
        plot_dpi (int, optional):         Pixels per inch to generate when saving the file.
        plot_linewidth (float, optional): Linewidth to use for lines in each plot.
        plot_close (bool, optional):      Whether to close each plot after generation. Recommended to avoid excessive memory use.
    
    Returns: 
        No return value. Creates plots that can be later displayed using plt.show(). Also saves .png files if desired.
    '''
    from pandas.plotting import register_matplotlib_converters
    register_matplotlib_converters()

    if save_plots:
        save_path_full = os.path.join( save_path, filename_header)
        if not os.path.isdir(save_path_full):
            os.mkdir( save_path_full)

    if ('session' in all_labels_dataframe.columns) and ('session' not in plot_skip):
        print('Plotting datetime check')
        fig_handle = plt.figure(figsize=plot_size, dpi=plot_dpi)
        plt.title('Datetime Validity Check')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['session'], linewidth=plot_linewidth)
        plt.xlabel('Date Time')
        plt.ylabel('Session')

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_datetime.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_datetime.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('event_startStop' in all_labels_dataframe.columns) and ('event_startStop' not in plot_skip):
        print('Plotting startStop')
        fig_handle = plt.figure(figsize=plot_size, dpi=plot_dpi)
        plt.title('Seizure Start / Stop')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['event_startStop'], linewidth=plot_linewidth)
        plt.xlabel('Date Time')
        plt.ylabel('Label (1 = Start, -1 = Stop)')

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_startStop.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_startStop.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('time_beforeEvent' in all_labels_dataframe.columns) and ('time_beforeEvent' not in plot_skip):
        print('Plotting time before event')
        fig_handle = plt.figure(figsize=plot_size, dpi=plot_dpi)
        plt.title('Time Before Event')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['time_beforeEvent'], linewidth=plot_linewidth)
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['label']*max(all_labels_dataframe['time_beforeEvent']), linewidth=plot_linewidth )
        plt.xlabel('Date Time')
        plt.ylabel('Seconds')

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_timeBeforeEvent.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_timeBeforeEvent.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('time_afterEvent' in all_labels_dataframe.columns) and ('time_afterEvent' not in plot_skip):
        print('Plotting time after event')
        fig_handle = plt.figure(figsize=plot_size, dpi=100)
        plt.title('Time After Event')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['time_afterEvent'], linewidth=plot_linewidth)
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['label']*max(all_labels_dataframe['time_afterEvent']), linewidth=plot_linewidth )
        plt.xlabel('Date Time')
        plt.ylabel('Seconds')

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_timeAfterEvent.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_datetime.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('time_duringEvent' in all_labels_dataframe.columns) and ('time_duringEvent' not in plot_skip):
        print('Plotting time since event start')
        fig_handle = plt.figure(7, figsize=plot_size, dpi=100)
        plt.title('Thing Progress')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['time_duringEvent'], linewidth=plot_linewidth)
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['label']*max(all_labels_dataframe['time_duringEvent']), linewidth=plot_linewidth )
        plt.xlabel('Date Time')
        plt.ylabel('Seconds')
        
        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_timeDuringEvent.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_timeDuringEvent.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('labels' not in plot_skip):
        print('Plotting label values')
        fig_handle = plt.figure(figsize=plot_size, dpi=plot_dpi)
        plt.title('Data Labels')
        plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['label']+0.04, 'b',  linewidth=plot_linewidth)

        if 'beforeEvent' in all_labels_dataframe.columns:
            index_plot = all_labels_dataframe['beforeEvent'] > 0
            plt.plot( all_labels_dataframe.loc[index_plot, col_x_axis], all_labels_dataframe.loc[index_plot,  'beforeEvent']+0.02, 'g.', linewidth=plot_linewidth)
        if 'afterEvent' in all_labels_dataframe.columns:
            index_plot = all_labels_dataframe['afterEvent'] > 0
            plt.plot( all_labels_dataframe.loc[index_plot, col_x_axis], all_labels_dataframe.loc[index_plot,   'afterEvent']-0.02, 'm.', linewidth=plot_linewidth)
        if 'betweenEvent' in all_labels_dataframe.columns:
            index_plot = all_labels_dataframe['betweenEvent'] > 0
            plt.plot( all_labels_dataframe.loc[index_plot, col_x_axis], all_labels_dataframe.loc[index_plot, 'betweenEvent']-0.04, 'r.', linewidth=plot_linewidth)
        plt.xlabel('Date Time')
        plt.ylabel('Label '+str(label_key))

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_labels.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_labels.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()


    if ('label_int' in all_labels_dataframe.columns) and ('label_int' not in plot_skip):
        print('Plotting label')
        fig_handle = plt.figure(figsize=plot_size, dpi=plot_dpi)
        plt.title('Data Labels (Condensed)')
        plt.plot(  all_labels_dataframe[col_x_axis], all_labels_dataframe['label_int'])
        plt.xlabel('Date Time')
        plt.ylabel('Label '+str(label_key))

        if save_plots:
            plt.savefig(os.path.join( save_path_full, filename_header) + '_labels-int.png')
            if save_pkl:
                with open( os.path.join( save_path_full, filename_header) + '_labels-int.pkl', 'wb') as f:
                    pickle.dump( fig_handle, f)
        if plot_close:    
            plt.clf()
            plt.close()





[docs]def plot_segments( all_labels_dataframe, segments, sample_rate, save_plots, save_path, filename_header, col_x_axis='datetime_ts', highlight_border=0.5, highlight_colors=['y','b','g','m','c','r'], save_pkl=False, plot_size=[16,9], plot_dpi=100, plot_linewidth=1, plot_close=True):
    '''General funcitons used for plotting output to check the resulting labeling data.

    Args:
        all_labels_dataframe (dataframe):   Dataframe containing all raw data
        segments (list):                    List of dictionaries defining segments. See CID convention for this dictionary in toast/bread/README.md
        sample_rate (int):                  Rate that data is sampled at.
        save_plots (bool):                  Whether to save the plots
        save_path (str):                    Path to save the data (if save_plots is true)
        filename_header (str):              Prefix to use for filename to identify the contents of the plot
        col_x_axis (str, optional):         Column in all_labels_dataframe to use for the x axis.
        highlight_border (float, optional): Number of seconds to leave blank on either side when shading samples. This helps visually separate each sample.
        highlight_colors (list, optional):  list of colors to use when highlighting each label.
        save_pkl (bool, optional):          Whether to save figure in a pickle file.  Note these files can be large files (>1GB).
        plot_size (list, optional):         Dimensions of each output figure
        plot_dpi (int, optional):           Pixels per inch to generate when saving the file.
        plot_linewidth (float, optional):   Linewidth to use for lines in each plot.
        plot_close (bool, optional):        Whether to close each plot after generation. Recommended to avoid excessive memory use.

    Returns: 
        No return value. Creates plots that can be later displayed using plt.show(). Also saves .png files if desired.
    '''
    from pandas.plotting import register_matplotlib_converters
    register_matplotlib_converters()

    if save_plots:
        save_path_full = os.path.join( save_path, filename_header)
        if not os.path.isdir(save_path_full):
            os.mkdir( save_path_full)

    fig_handle = plt.figure(6, figsize=plot_size, dpi=plot_dpi)
    plt.plot( all_labels_dataframe[col_x_axis], all_labels_dataframe['label_int'], linewidth=plot_linewidth)
    plt.title('Segment Summary')

    data_continuity_gaps = []
    for index, _ in all_labels_dataframe[ all_labels_dataframe['datetime_ts'].diff() > 2/sample_rate ].iterrows():
        data_continuity_gaps.append([ all_labels_dataframe.loc[index-1, col_x_axis], all_labels_dataframe.loc[index, col_x_axis] ])
    for gap in data_continuity_gaps:
        plt.plot( gap, [0, 0], 'r', linewidth=plot_linewidth*3)

    for label in segments:
        for segment in label:
            if len(segment):
                try:
                    seg_start = segment['start_'+col_x_axis] + highlight_border
                    seg_stop  = segment['stop_'+col_x_axis]  - highlight_border
                except:
                    seg_start = segment['start_datetime'].timestamp() + highlight_border
                    seg_stop  = segment[ 'stop_datetime'].timestamp()  - highlight_border

                plt.axvspan(seg_start, seg_stop, color=highlight_colors[ segment['label_int']], alpha=0.3)

    if save_plots:
        print('Saving segments')
        plt.savefig(os.path.join( save_path_full, filename_header) + '_segments.png')
        if save_pkl:
            with open( os.path.join( save_path_full, filename_header) + '_segments.pkl', 'wb') as f:
                pickle.dump( fig_handle, f)
    if plot_close:    
        plt.clf()
        plt.close()
