    if accuracy_data and (len(accuracy_data) == len(testing_data)):
        best_accuracy = argmax( accuracy_data)
        worst_accuracy = argmin( accuracy_data)

            if accuracy_data:

        if not os.path.exists(os.path.join( save_path, output_folder_name)):

        x_axis (iterable):                      1D Iterable: Variable agains which to plot accuracy. Typically Epochs

        accuracy_data (iterable):               1D Iterable: Testing Accuracy, ordered according to values in x_axis

        false_neg (iterable, optional):         1D Iterable: False Negative rate, ordered according to values in x_axis

        y_label (str -or- list, optional):      Use for y axis label. If sting will label testing_data axis, if list will label both testing_data and accuracy_data axis.
        title (str, optional):                  Use for title
        testing_label (str, optional):          Use in legend to identify training_data line
        accuracy_label (str, optional):         Use in legend to identify testing_data line
        show_plot (bool, optional):             Whether to render the plot to the screen
        save_plot (bool, optional):             Whether to save the plot as a .png
        save_path (str, optional):              Path to save plots to. Note plots will be saved to save_path + output_folder_name location
        output_folder_name (str, optional):     Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        description (str, optional):            A description of the data in the plot. This will be used as the filename.
        bkg_color (str -or- tuple, optional):   Character or RBG Thruple representing the background color of the plot.
        path_log (str, optional):               Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        close_plot (bool, optional):            Whether to close the figure when finished. THis helps minimize memory usage.
        use_agg (bool, optional):               Whether to use the Agg backend of matplotlib.  Use when connecting to non-display machines/servers

    Returns:
        No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    if use_agg:
        import matplotlib
        matplotlib.use('Agg')
    import matplotlib.pyplot as plt

    print('Plotting ', description, end='\r')
    # Plot the testing loss, mostly for reference to see if network was converging or diverging.
    _, ax0 = plt.subplots()
    ax0.set_facecolor(bkg_color)
    ax0.plot( x_axis, testing_data, 'go-', label=testing_label, alpha=0.5)
    # If testing_progress is included and matches testing data, plot the range of training loss (min to max) as a vertical line
    # Used to visualize the relationship of the final loss to the previous loss values, and identify general convergence/divergence of network.
    if testing_progress and (len(testing_progress) == len(testing_data)):
        for i, x in enumerate( x_axis):
            plt.plot( [x, x], testing_progress[i], 'g-')

    ax1 = ax0.twinx()
    plt.title(title)
    ax0.set_xlabel(x_label)
    if type(y_label) == list:
        ax0.set_ylabel(y_label[0])
        ax0.tick_params(axis='y', colors='g')
        ax1.set_ylabel(y_label[1])
        ax1.tick_params(axis='y', colors='m')
    else:
        ax0.set_ylabel(y_label)

    # Plot the accuracy_progress. Note this is plotted before accuracy_data so accuracy_data remains on top
    # If accuracy_progress is included and matches accuracy_data, plot the range of training loss (min to max) as a vertical line
    # Used to visualize the relationship of the final loss to the previous loss values, and identify general convergence/divergence of network.
    if accuracy_progress and (len(accuracy_progress) == len(accuracy_data)):
        for i, x in enumerate( x_axis):
            plt.plot( [x, x], accuracy_progress[i], 'm-')

    # If the false positive rate is included and matches accuracy data, plot this value as a red line ABOVE the accuracy plot
    if false_pos and (len(false_pos) == len(accuracy_data)):
        for i, x in enumerate( x_axis):
            plt.plot( [x, x], [accuracy_data[i], accuracy_data[i]+false_pos[i]], 'r-', alpha=0.5)

    # If the false negative rate is included and matches accuracy data, plot this value as a red line BELOW the accuracy plot
    if false_neg and (len(false_neg) == len(accuracy_data)):
        for i, x in enumerate( x_axis):
            plt.plot( [x, x], [accuracy_data[i], accuracy_data[i]-false_neg[i]], 'r-', alpha=0.5)

    # Plot the accuracy data
    ax1.plot( x_axis, accuracy_data,  'mo-', label=accuracy_label)
    plt.tight_layout()
    plt.legend()
    
    if show_plot:
        plt.show()
    if save_plot:
        if not os.path.exists(os.path.join( save_path, output_folder_name)):
            os.mkdir(os.path.join( save_path, output_folder_name))
        tlog.output_log('Saving : '+str(os.path.join( save_path, output_folder_name, description+'.png')), path_log)
        plt.savefig(os.path.join( save_path, output_folder_name, description+'.png'))
    if close_plot:
        plt.close()

    '''Plots histograms of values for each layer to visualize distribution of data in each layer.

    Args:  
        data (iterable):                      2D or 3D Iterable: 3D-values for successive layers, (optional)2D-values for different categories/outputs, 1D-values whose distribution will be displayed
        testing_data (iterable):              1D Iterable: Testing Loss, ordered according to values in x_axis
        bins (int, optional):                 How many bins to use in histogram. If none provided, will use twice the number of values in the second to last layer, or 100, whichever is smaller.
        save_txt (bool, optional):            Whether to save a .txt summary of the final state of the varaibles for each layer
        x_label (str, optional):              X axis label
        y_label (list, optional):             Label for each layer.  If nothing is provided, the layer number is used.
        title (str, optional):                Title of plot
        show_plot (bool, optional):           Whether to render the plot to the screen
        save_plot (bool, optional):           Whether to save the plot as a .png
        save_path (str, optional):            Path to save plots to. Note plots will be saved to save_path + output_folder_name location
        output_folder_name (str, optional):   Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        description (str, optional):          Description of the data in the plot. This will be used as the filename.
        bkg_color (str -or- tuple, optional): Character or RBG Thruple representing the background color of the plot.
        path_log (str, optional):             Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        close_plot (bool, optional):            Whether to close the figure when finished. THis helps minimize memory usage.
        use_agg (bool, optional):             Whether to use the Agg backend of matplotlib.  Use when connecting to non-display machines/servers

    Returns:
        No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    if use_agg:
        import matplotlib
        matplotlib.use('Agg')
    import matplotlib.pyplot as plt
    colors = ['m', 'c', 'b', 'r', 'g', 'y']
    
    print('Plotting ', description, end='\r')
    total_layers = len(data)
    if bins is None:
        try:
            bins = min( data[-2].shape[-1] * 2, 100)
        except:
            bins = None

    fig_dpi = 100 # default=100
    plt.figure(figsize=(9, 16), dpi=fig_dpi)

    # Plot data from each layer in a separate subplot
    for layer_num, layer_data in enumerate( data):
        ax = plt.subplot(total_layers, 1, layer_num+1)
        ax.set_facecolor(bkg_color)

        # If multiple dimensions exist for each layer, display each column of data in a different color
        if (len( layer_data.shape) > 1):
            for i, data_column in enumerate( layer_data.transpose()):
                plt.hist( data_column, bins=bins, color=colors[i%len(colors)], alpha=0.5)
        # If only one dimension exists for each layer, display the entire distribution in one color.
        else:
            plt.hist( layer_data, bins=bins)
            
        # if y_label does not exist, label it with the layer number.
        if y_label is not None:
            ax.set_ylabel( str(y_label[layer_num]))
        else:
            ax.set_ylabel(str(layer_num))
        ax.set_yticklabels([])
        # ax.set_xticklabels/([])
        if layer_num == 0 and title is not None:
            plt.title(title)
        # To save space, only plot the x_label on the bottom plot
        if layer_num == total_layers - 1:
            plt.xlabel(x_label)

    if save_txt:
        # Calculate some general statistics about the distribution of the data in each layer
        avg       = []
        stdev     = []
        q1_value  = []
        q3_value  = []
        max_value = []
        min_value = []
        for layer in data:
            if (len( layer_data.shape) > 1):
                avg_list = []
                std_list = []
                q1_list  = []
                q3_list  = []
                max_list = []
                min_list = []
                for data_column in layer.transpose():
                    avg_list.append( mean(layer))
                    std_list.append( std(layer))
                    q1_list.append(  percentile( data_column, 25))
                    q3_list.append(  percentile( data_column, 75))
                    max_list.append( data_column[ argmax(data_column)])
                    min_list.append( data_column[ argmin(data_column)])
                avg.append( avg_list)
                stdev.append( std_list)
                q1_value.append(  q1_list)
                q3_value.append(  q3_list)
                max_value.append( max_list)
                min_value.append( min_list)
            else:
                avg.append(   mean(layer))
                stdev.append(  std(layer))
                q1_value.append( percentile( layer, 25))
                q3_value.append( percentile( layer, 75))
                max_value.append(    layer[ argmax(layer)])
                min_value.append(  layer[ argmin(layer)])

        # Write the values of each layer into a text file.
        if not os.path.exists(os.path.join( save_path, output_folder_name)):
            os.mkdir(os.path.join( save_path, output_folder_name))
        print('Saving : ', os.path.join( save_path, output_folder_name, description+'.txt'))
        with open( os.path.join( save_path, output_folder_name, description+'.txt'), 'w') as f:
            layer = 0
            for a, s, qI, qIII, b, c in zip(avg, stdev, q1_value, q3_value, max_value, min_value):
                try:
                    f.write('Layer {:2d} parameters:\tmean: {:.6e}\tstdev: {:.6e}\tquartile 1: {:.6e}\tquartile 3: {:.6e}\tmax: {:.6e}\tmin: {:.6e}\n'.format( layer, a, s, qI, qIII, b, c))
                except:
                    f.write('Layer {} parameters:\tmean: {}\tstdev: {}\tquartile 1: {}\tquartile 3: {}\tmax: {}\tmin: {}\n'.format( layer, a, s, qI, qIII, b, c))
                layer += 1

    if show_plot:
        plt.show()
    if save_plot:
        if not os.path.exists(os.path.join( save_path, output_folder_name)):
            os.mkdir(os.path.join( save_path, output_folder_name))

        plt.savefig(os.path.join( save_path, output_folder_name, description+'.png'))

        plt.close()

    '''Visualize the distribution of many parallel variables during training.  
    Typically used to track and plot layer weight or activation distributions throughout training.

    Args:  
        x_axis (iterable):                   1D Iterable: Variable agains which to plot avg_data. Typically Epochs
        avg_data (iterable):                 2D Iterable: 1D-values for each layer, 2D-averages of data to track, ordered according to values in x_axis
        std_data (iterable, optional):       2D Iterable: 1D-values for each layer, 2D-stdev of data to track, ordered according to values in x_axis
        q1_data (iterable, optional):        2D Iterable: 1D-values for each layer, 2D-first quartile of data to track, ordered according to values in x_axis
        q3_data (iterable, optional):        2D Iterable: 1D-values for each layer, 2D-third quartile of data to track, ordered according to values in x_axis
        min_data (iterable, optional):       2D Iterable: 1D-values for each layer, 2D-minimum of data to track, ordered according to values in x_axis
        max_data (iterable, optional):       2D Iterable: 1D-values for each layer, 2D-maximum of data to track, ordered according to values in x_axis
        num_std_bars (int, optional):        Maximum number of standard deviation error bars to plot. Is limited to allow the shaded quartile data to also be visible.
        x_label (str, optional):             X axis label. Note that y_label is automatically set to the layer number
        title (str, optional):               Title of plot
        show_plot (bool, optional):          Whether to render the plot to the screen
        save_plot (bool, optional):          Whether to save the plot as a .png
        save_path (str, optional):           Path to save plots to. Note plots will be saved to save_path + output_folder_name location
        output_folder_name (str, optional):  Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        description (str, optional):         Description of the data in the plot. This will be used as the filename.
        bkg_color (st -or- tuple, optional): Character or RBG Thruple representing the background color of the plot.
        path_log (str, optional):            Folder name to save plots into. Note plots will be saved to save_path + output_folder_name location
        close_plot (bool, optional):            Whether to close the figure when finished. THis helps minimize memory usage.
        use_agg (bool, optional):            Whether to use the Agg backend of matplotlib.  Use when connecting to non-display machines/servers

    Returns:
        No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    if use_agg:
        import matplotlib
        matplotlib.use('Agg')
    import matplotlib.pyplot as plt

    print('Plotting ', description, end='\r')
    fig_dpi = 100 # default=100
    plt.figure(figsize=(9, 16), dpi=fig_dpi)

    num_layers = len(avg_data)
    # Plot data from each layer in a separate subplot
    for layer_num in range( num_layers):
        ax0 = plt.subplot( num_layers, 1, layer_num+1)
        ax0.set_facecolor(bkg_color)
        plt.axhline(0, color='grey')
        plt.plot( x_axis, avg_data[layer_num], color='b')
        # Error bars represent standard deviation.  Only plot num_std_bars standard dev so you can see cyan quartile coloring
        if std_data is not None:
            if len(x_axis) > num_std_bars:
                std_x = [int(i) for i in linspace(0, len(x_axis)-1, num_std_bars)]
                std_y = [std_data[layer_num][i] for i in std_x]
            else:
                std_x = x_axis
                std_y = std_data[layer_num]
            for std_var_x, std_var_y, avg_val_y in zip( std_x, std_y, avg_data[layer_num]):
                plt.plot([std_var_x, std_var_x], [avg_val_y-std_var_y, avg_val_y+std_var_y], 'b')
        # Cyan filled area represents the first and third quartile
        if q1_data is not None:
            plt.fill_between( x_axis, q1_data[layer_num], avg_data[layer_num], color='c')
        if q3_data is not None:
            plt.fill_between( x_axis, avg_data[layer_num], q3_data[layer_num], color='c')
        # Green line represents the min and max values
        if min_data is not None:
            plt.plot( x_axis, min_data[layer_num], color='g')
        if max_data is not None:
            plt.plot( x_axis, max_data[layer_num], color='g')
        
        plt.ylabel(layer_num)
        if (layer_num == 0) and (title is not None):
            plt.title(title)
        if (layer_num == num_layers-1):
            plt.xlabel(x_label)

        # Try to center the y axis around the ending values, so that the data isn't hard to see as a result of a large change.
        try:
            data_range = max_data[layer_num][-1] - min_data[layer_num][-1]
            lower_limit = min_data[layer_num][-1] - data_range*0.5
            upper_limit = max_data[layer_num][-1] + data_range*0.5
            # If the upper and lower limit are the same it throws a warning. This is the only way to silence that warning right now.
            counter_loop = 0
            while (upper_limit == lower_limit) and (counter_loop < 1e6):
                lower_limit  *= 0.95
                counter_loop += 1
            plt.ylim((lower_limit, upper_limit))
        except:
            try:
                data_range = q3_data[layer_num][-1] - q1_data[layer_num][-1]
                lower_limit = min_data[layer_num][-1] - data_range*0.5
