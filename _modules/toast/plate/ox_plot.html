    '''Plots all 4 baseline channels Respiration, EKG1, EKG2, and ECoG

        Args:
            time (iterable):                  Time array of data
            data (numpy array):               Numpy ndarray of data with row0=resp, row1=EKG1, row2=EKG2, row3=ECoG
            start_seizure (list, optional):   List of times that labeled seizures start. Units should match time array.
            end_seizure (list, optional):     List of times that labeled seizures end.   Units should match time array.
            plot_linewidth (float, optional): Linewidth used by matplotlib to plot data.
            title (str, optional):            Title of plot
            save_plots (bool, optional):      Whether plots should be saved.
            save_path (str, optional):        Path of folder for file to be saved in
            save_filename (str, optional):    Filename (without extnesion) to save file under.
            save_dpi (int, optional):          Number of dots per inch in the output figure. Controls resolution.
            save_pkl (bool, optional):        Whether to save a pickle file of the plot (not recommended).

        Returns:
            No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''

    figures = []
    figures.append( plt.figure())
    ax0 = plt.subplot(4, 1, 1)
    ax1 = plt.subplot(4, 1, 2, sharex=ax0)
    ax2 = plt.subplot(4, 1, 3, sharex=ax0)
    ax3 = plt.subplot(4, 1, 4, sharex=ax0)
    ax3.set_xlabel('Time (s)')
    if (start_seizure is not None) and (end_seizure is not None):
        for i in range( len( start_seizure)):
            ax0.axvspan( start_seizure[i], end_seizure[i], color='m', alpha=0.2)
            ax1.axvspan( start_seizure[i], end_seizure[i], color='m', alpha=0.2)
            ax2.axvspan( start_seizure[i], end_seizure[i], color='m', alpha=0.2)
            ax3.axvspan( start_seizure[i], end_seizure[i], color='m', alpha=0.2)
    
    ax0.plot( time, data[0, :], linewidth=plot_linewidth)
    ax0.set_ylabel('Respiration')
    ax1.plot( time, data[1, :], linewidth=plot_linewidth)
    ax1.set_ylabel('EKG_1')
    ax2.plot( time, data[2, :], linewidth=plot_linewidth)
    ax2.set_ylabel('EKG_2')
    ax3.plot( time, data[3, :], linewidth=plot_linewidth)
    ax3.set_ylabel('ECoG')
    if title is not None:
        ax0.set_title(title)
    plt.tight_layout()

    if save_plots:
        save_path_full = os.path.join( save_path, save_filename)

        for i, fig_handle in enumerate( figures):
            print('Saving figure ', i)
            plt.savefig(save_path_full + '.png', dpi=save_dpi)
            if save_pkl:
              with open( save_path_full + '.pkl', 'wb') as f:
                pickle.dump( fig_handle, f)

    '''Plots all 4 fft output of channels Respiration, EKG1, EKG2, and ECoG

        Args:  
            data (numpy array):               Numpy ndarray of data where  row0=respiration, row1=EKG1, row2=EKG2, row3=ECoG
            fs (int):                         Sample Frequency (Hz)
            plot_linewidth (float, optional): Linewidth used by matplotlib to plot data.
            title (str, optional):            Title of plot
            save_plots (bool, optional):      Whether plots should be saved.
            save_path (str, optional):        Path of folder for file to be saved in
            save_filename (str, optional):    Filename (without extnesion) to save file under.
            save_dpi (int, optional):          Number of dots per inch in the output figure. Controls resolution.
            save_pkl (bool, optional):        Whether to save a pickle file of the plot (not recommended)

        Returns:
            No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    from toast.butter.freq import fourier
    figures = []

    print('Getting FFT')
    plot_freq, plot_channels_fft = fourier.get_fft( data, fs)
    
    print('Plotting')
    figures.append( plt.figure())
    ax0 = plt.subplot( 4, 1, 1)
    ax1 = plt.subplot( 4, 1, 2, sharex=ax0)
    ax2 = plt.subplot( 4, 1, 3, sharex=ax0)
    ax3 = plt.subplot( 4, 1, 4, sharex=ax0)
    ax3.set_xlabel('Freq (Hz)')

    ax0.plot( plot_freq, plot_channels_fft[0], linewidth=plot_linewidth)
    ax0.set_ylabel('Respiration')
    ax1.plot( plot_freq, plot_channels_fft[1], linewidth=plot_linewidth)
    ax1.set_ylabel('ECG_1')
    ax2.plot( plot_freq, plot_channels_fft[2], linewidth=plot_linewidth)
    ax2.set_ylabel('ECG_2')
    ax3.plot( plot_freq, plot_channels_fft[3], linewidth=plot_linewidth)
    ax3.set_ylabel('ECoG')

    plt.xlim(0, max(plot_freq))
    if title is not None:
        ax0.set_title(title)
    plt.tight_layout()

    if save_plots:
        save_path_full = os.path.join( save_path, save_filename)

        for i, fig_handle in enumerate(figures):
            print('Saving figure ', i)
            plt.savefig(save_path_full + '.png', dpi=save_dpi)
            if save_pkl:
              with open( save_path_full + '.pkl', 'wb') as f:
                pickle.dump( fig_handle, f)




[docs]def show_stft( time, data, fs, window, step, zero_extend=1, scaling=1, normalize=False, exclude_offset=[[]], frequency_range=[0, 150], start_marker=[], end_marker=[], title=None, save_plots=False, save_path='./', save_filename='all_stft', save_dpi=100, save_pkl=False):
    '''Plots the stft of the given data. Seizures are marked on the X axis (Green=start, Red=end)

        Arguments marked with an '*' are passed directly to :py:func:`~toast.bread.format.save_data.save_to_some_file` .
        Refer to this function definition for full functionality description.

        Note:
            This function can only plot single-channel data.

        Args:
            time (iterable):                   Time array of data
            data (numpy array):                Numpy array or ndarray (single row) representing the data to be plotted.
            fs (int):                          Sample rate of the data.
            window (int):                     Size of the window used when computing the fft for the stft
            step (int):                       Step size to use when sliding widow across data.
            zero_extend (int, optional):      Multiple of len(data) to zero extend by.  i.e. passing 2 would add 2*len(data) zeros to the end of each window of data.
            scaling (int, optional):          Integer level for relative scaling of coefficients (for plotting).
            exclude_offset (list, optional):  2D list of time offsets to exclude for plotting.  Values in this range are set to the minimum existing coefficient value.
            frequency_range (list, optional): Frequency range to return, if only interested in a specific range. Note the full stft is calculated regardless.
            start_marker (list, optional):     1D List of times in which labeled seizures start. Units should match time array.
            end_marker (list, optional):       1D List of times in which labeled seizures end.   Units should match time array.
            title (str, optional):             Title of plot
            save_plots (bool, optional):       Whether plots should be saved.
            save_path (str, optional):         Path of folder for file to be saved in
            save_filename (str, optional):     Filename (without extnesion) to save file under.
            save_dpi (int, optional):          Number of dots per inch in the output figure. Controls resolution.
            save_pkl (bool, optional):         Whether to save a pickle file of the plot (not recommended)

        Returns:
            No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    from toast.butter.freq import fourier
    from matplotlib import ticker, cm
    figures = []

    print('Getting STFT')
    time_stft, freq, stft_values = fourier.get_stft( data, fs, window, step, zero_extend=zero_extend, scaling=scaling, normalize=normalize, exclude_offset=exclude_offset, frequency_range=frequency_range)
    time_stft += time[0]

    print('Plotting')
    fig, ax = plt.subplots()
    figures.append( fig)
    cs = ax.contourf(time_stft, freq, stft_values, cmap='gnuplot')
    # cs = ax.contourf(x, y, z, locator=ticker.LogLocator(), cmap='plasma')
    _ = fig.colorbar(cs)

    if (len(start_marker) > 0) or (len(end_marker) > 0):
        plt.xlim((time_stft[0], time_stft[-1]))
        ticks = list( plt.xticks()[0])[1:-1]
        ticks = [time_stft[0]] + ticks
        ticks.append( int( time_stft[-1]))
        tick_colors = ['k'] * len(ticks)
        tick_rotation = [0] * len(ticks)
        for i in range( len(start_marker)):
            ticks.append( end_marker[i][0])
            tick_colors.append( 'r')
            tick_rotation.append(90)
            ticks.append(  start_marker[i][0])
            tick_colors.append( 'g')
            tick_rotation.append(90)
        plt.xticks( ticks)
        for ticklabel, tickcolor, tickrotation in zip(plt.gca().get_xticklabels(), tick_colors, tick_rotation):
            ticklabel.set_color(tickcolor)
            ticklabel.set_rotation(tickrotation)

    plt.ylabel('Frequency (Hz)')
    plt.xlabel('Time (sec)')
    if title is not None:
        plt.title(title)
    plt.tight_layout()

    if save_plots:
        save_path_full = os.path.join( save_path, save_filename)

        for i, fig_handle in enumerate(figures):
            print('Saving figure ', i)
            plt.savefig(save_path_full + '.png', dpi=save_dpi)
            if save_pkl:
              with open( save_path_full + '.pkl', 'wb') as f:
                pickle.dump( fig_handle, f)
