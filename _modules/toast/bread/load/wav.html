'''
    All files here include the same set or subset of functions used to load raw data into a program.

    `*` represents interchangably either the `label`, `data`, or `metadata` suffix.

    =====================  ============
    Function Name          Description
    =====================  ============
    ``import_*``           import an entire data file (session) into a script.
    ``recall_*``           import only a portion of a data file into a script (segment or sample).
    ``recall_*_channels``  import a portion of the data file and shape it inot a numpy array where each row in a data channel and each column is a time point.
    =====================  ============


    See the individual sripts for specifics on arguments and return values as they vary slightly depending on the file type.
'''

from os.path import join

[docs]def import_data(file_path, filename, dataset=None, datatype=np.int16):
    '''Imports all data in the wav file

    Args:
        file_path (str):         Path to the folder where the data exists. Separated for easier iteration through a file tree.
        filename (str):          The specific filename of the file to process.  Do not include extensions.
        dataset (str, optional): Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        datatype (np dtype, optional): Data type to use when unpacking the data

    Returns:
        Numpy array (flattened) of the data, number of channels
    '''
    if dataset is None:
        path = join(file_path, filename) + '_data.wav'
    else:
        path = join(file_path, filename) + '_data-' + dataset + '.wav'
    # fs, data = scipy.io.wavfile.read(filename + '_data.wav')
    with wave.open( path, 'rb') as f:
        data = np.frombuffer( f.readframes( f.getnframes()), dtype=datatype)
        number_channels = f.getnchannels()
    return data, number_channels

    '''Imports some data in the wav file, starting at frame `start` and stopping at frame `stop` (includes start, excludes stop)

    Args:
        file_path (str):         Path to the folder where the data exists. Separated for easier iteration through a file tree.
        filename (str):          The specific filename of the file to process.  Do not include extensions.
        start (int):             Frame to start importing (inclusive)
        stop (int):              Frame to stop importing (exclusive)
        dataset (str, optional): Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        datatype (np dtype, optional): Data type to use when unpacking the data

    Returns:
        Numpy array (flattened) of the data, number of channels
    '''
    if dataset is None:
        path = join(file_path, filename) + '_data.wav'
    else:
        path = join(file_path, filename) + '_data-' + dataset + '.wav'
    with wave.open( path, 'rb') as f:
        f.setpos( start)
        data = np.frombuffer( f.readframes( (stop-start)), dtype=datatype)
        number_channels = f.getnchannels()
    return data, number_channels

[docs]def recall_data_fs(file_path, filename, start, stop, dataset=None, datatype=np.int16):
    '''Imports some data in the wav file, starting at frame `start` and stopping at frame `stop` (includes start, excludes stop)

    This is separated from ``~recall_data`` because it has to extract and return an additional parameter. ``~recall_data`` is kept extremely optimized for use in ML pipelines.

    Args:
        file_path (str):         Path to the folder where the data exists. Separated for easier iteration through a file tree.
        filename (str):          The specific filename of the file to process.  Do not include extensions.
        start (int):             Frame to start importing (inclusive)
        stop (int):              Frame to stop importing (exclusive)
        dataset (str, optional): Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        datatype (np dtype, optional): Data type to use when unpacking the data

    Returns:
        Numpy array (flattened) of the data, number of channels
    '''
    if dataset is None:
        path = join(file_path, filename) + '_data.wav'
    else:
        path = join(file_path, filename) + '_data-' + dataset + '.wav'
    with wave.open( path, 'rb') as f:
        f.setpos( start)
        data = np.frombuffer( f.readframes( (stop-start)), dtype=datatype)
        number_channels = f.getnchannels()
        sample_rate = f.getframerate()
    return data, number_channels, sample_rate

[docs]def recall_data_channels(file_path, filename, start, stop, dataset=None, datatype=np.int16):
    '''Imports some data in the wav file, starting at frame `start` and stopping at frame `stop` (includes start, excludes stop).
    
    Reshapes data into a square with each row coresponding to a data channel.  Also calculates the corresponding time array.

    Args:
        file_path (str):         Path to the folder where the data exists. Separated for easier iteration through a file tree.
        filename (str):          The specific filename of the file to process.  Do not include extensions.
        start (int):             Frame to start importing (inclusive)
        stop (int):              Frame to stop importing (exclusive)
        dataset (str, optional): Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        datatype (np dtype, optional): Data type to use when unpacking the data

    Returns:
        Numpy array of the data with each channel accross a row, corresponding time array
    '''
    data, n_channels, fs = recall_data_fs( file_path, filename, start, stop, dataset=dataset, datatype=datatype)
    channels = np.reshape(data, (-1, n_channels)).transpose()
    time = np.linspace(start/fs, (start+channels.shape[1])/fs, channels.shape[1])
    return channels, time
