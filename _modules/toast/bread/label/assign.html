    '''Adds several columns to a dataframe that includes pointwise labels (integer form) relating to some boolean event in ``column_event``

    Within the passed dataframe in column ``column_event``, this function assuems 

        - label 0 is a null label
        - label 1 is the event of interest    
    
    The following columsn will be added to the dataframe
    
        * `event_startStop` (int), +1 = start; -1 = end
        * `time_beforeEvent` (float), seconds before the next event
        * `time_afterEvent` (float), seconds after the end of the last event
        * `time_duringEvent` (float), time since the start of the event. Only >0 during the event.
        * `beforeEvent` (int), indicates a valid beforeEvent time based on inputs. 0 = not valid, ``beforeEvent_label`` = valid
        * `afterEvent` (int), indicates a valid afterEvent time based on inputs. 0 = not valid, ``afterEvent_label`` = valid
        * `betweenEvent` (int), indicates a valid betweenEvent time based on inputs. 0 = not valid, ``betweenEvent_label`` = valid
        * `label_int` (int), combines all of the above labels. Takes the maximum label value for each time point.
    
    Tip:
        This function will create multiple columns in the passed dataframe.
        To delete a column you may not need from a dataframe: ``all_labels_dataframe.drop(['event_startStop', 'prething', ...], axis=1)``

    Args:   
        all_labels_dataframe (dataframe):   Dataframe with all of the data and labels in a column named "thing"
        time_beforeEvent (float):           Time (seconds) before an event (thing) that designates data as occuring right before a thing happens.
        time_afterEvent (float):            Time (seconds) after a thing that designates data as occuring right after a thing happens.
        time_betweenEvent_before (float):   Time (seconds) before an event when data can safely be labeles as a baseline between events.
        time_betweenEvent_after (float):    Time (seconds) after an event when data can safely be labeles as a baseline between events.
        column_event (str, optional):          Column in ``all_labels_dataframe`` where the event of interest is indicated.
        beforeEvent_label (int, optional):  Integer assigned to prething label
        afterEvent_label (int, optional):   Integer assigned to postthing label (can be the same as prething_label to combine labels.)
        betweenEvent_label (int, optional): Integer assigend to interthing label.
        path_log (str, optional):           Path to the file for outputing logs during execution.

    Returns: 
        A dataframe containing new labels relative to the passed "thing" label column.
    '''
    tlog.output_log('Finding '+str(column_event)+' start / stop times', path_log)
    all_labels_dataframe['event_startStop'] = all_labels_dataframe[column_event].diff()

    tlog.output_log('Calculating time to '+str(column_event), path_log)
    times_event_start = list( all_labels_dataframe[ all_labels_dataframe['event_startStop'] == 1]['datetime_ts'])

    all_labels_dataframe['time_beforeEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']
    for start_ts in sorted( times_event_start):
        all_labels_dataframe['time_beforeEvent'] = pd.concat( [all_labels_dataframe['time_beforeEvent'], start_ts-all_labels_dataframe['datetime_ts']], axis=1).min(axis=1)
        all_labels_dataframe.loc[ (all_labels_dataframe['time_beforeEvent'] < 0), 'time_beforeEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']

    tlog.output_log('Calculating time since '+str(column_event), path_log)
    times_event_end = list( all_labels_dataframe[ all_labels_dataframe['event_startStop'] == -1]['datetime_ts'])

    all_labels_dataframe['time_afterEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']
    for end_ts in sorted( times_event_end, reverse=True):
        all_labels_dataframe['time_afterEvent'] = pd.concat( [all_labels_dataframe['time_afterEvent'], all_labels_dataframe['datetime_ts']-end_ts], axis=1).min(axis=1)
        all_labels_dataframe.loc[ (all_labels_dataframe['time_afterEvent'] < 0), 'time_afterEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']

    print('Calculating event progress')
    all_labels_dataframe['time_duringEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']
    for start_ts in sorted( times_event_start, reverse=True):
        all_labels_dataframe['time_duringEvent'] = pd.concat( [all_labels_dataframe['time_duringEvent'], all_labels_dataframe['datetime_ts']-start_ts], axis=1).min(axis=1)
        all_labels_dataframe.loc[ (all_labels_dataframe['time_duringEvent'] < 0), 'time_duringEvent'] = all_labels_dataframe.loc[len(all_labels_dataframe)-1, 'datetime_ts']
    # Metric only valid during event.  If thing not occuring, set to "invalid" -1
    all_labels_dataframe.loc[ (all_labels_dataframe[column_event] == 0), 'time_duringEvent'] = -1

    # Determine beforeEvent period
    tlog.output_log('Determining beforeEvent Labels', path_log)
    all_labels_dataframe['beforeEvent'] = 0
    all_labels_dataframe.loc[ (all_labels_dataframe['time_beforeEvent'] < time_beforeEvent*60), 'beforeEvent'] = beforeEvent_label
    all_labels_dataframe.loc[ (all_labels_dataframe[column_event] == 1), 'beforeEvent'] = 0

    # Determine afterEvent period
    tlog.output_log('Determining afterEvent Labels', path_log)
    all_labels_dataframe['afterEvent'] = 0
    all_labels_dataframe.loc[ (all_labels_dataframe['time_afterEvent']  < time_afterEvent*60),  'afterEvent']  = afterEvent_label
    all_labels_dataframe.loc[ (all_labels_dataframe[column_event] == 1), 'afterEvent'] = 0

    # Determine valid inbetween periods
    tlog.output_log('Determining betweenEvent Labels')
    all_labels_dataframe['betweenEvent'] = 0
    all_labels_dataframe.loc[ ((all_labels_dataframe['time_beforeEvent'] > time_betweenEvent_before*3600) & (all_labels_dataframe['time_afterEvent'] > time_betweenEvent_after*3600)), 'interictal'] = betweenEvent_label
    all_labels_dataframe.loc[ (all_labels_dataframe[column_event] == 1), 'betweenEvent'] = 0
    # Exclude Ends
    allData_start = all_labels_dataframe.iloc[ 0]['datetime_ts']
    allData_end   = all_labels_dataframe.iloc[-1]['datetime_ts']
    all_labels_dataframe.loc[ ((all_labels_dataframe['datetime_ts'] - allData_start) < time_betweenEvent_after*3600),  'betweenEvent'] = 0
    all_labels_dataframe.loc[ ((allData_end - all_labels_dataframe['datetime_ts'])   < time_betweenEvent_before*3600), 'betweenEvent'] = 0

    tlog.output_log('Condensing labels to 1 column', path_log)
    all_labels_dataframe['label_int'] = all_labels_dataframe[[column_event, 'beforeEvent', 'afterEvent', 'betweenEvent']].max(axis=1)

    return all_labels_dataframe
