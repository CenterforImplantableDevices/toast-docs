from datetime import datetime



[docs]def get_timestamp():
    ''' Returns the current time as a timestamp string in a standardized format.
    All timestamps referenced in `toast` should use this function to maintain consistency.

    Timestamp format appears as: ``YYYY.MM.DD.HH.MM.SS``

    Args:
        None
    
    Returns:
        String representing the current time.
    '''
    ts = datetime.now()
    return "{:04d}.{:02d}.{:02d}.{:02d}.{:02d}.{:02d}".format(ts.year, ts.month, ts.day, ts.hour, ts.minute, ts.second)


    

[docs]def get_datetime( timestamp_str):
    ''' Returns the timestamp string as a datetime object

    Timestamp format appears as: ``YYYY.MM.DD.HH.MM.SS``

    Args:
        timestamp_str (str): Timestamp string to interpret
    
    Returns:
        datetime object matching the passed timestamp
    '''
    sep = [int(i) for i in timestamp_str.split('.')]
    return datetime(sep[0], sep[1], sep[2], sep[3], sep[4], sep[5])




[docs]def get_datetime_ms( timestamp_str):
    ''' Returns the timestamp string as a datetime object. Partial seconds are converted to millisecond unit.

    Timestamp format appears as: ``YYYY.MM.DD.HH.MM.SS.mmm``

    Args:
        timestamp_str (str): Timestamp string to interpret
    
    Returns:
        datetime object matching the passed timestamp
    '''
    sep = [int(i) for i in timestamp_str.split('.')]
    return datetime(sep[0], sep[1], sep[2], sep[3], sep[4], sep[5], int(sep[6]*1e3))
