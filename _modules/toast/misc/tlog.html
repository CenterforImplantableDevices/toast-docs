from toast.misc import ttime


[docs]def output_log( message, file_path='./log.txt'):
    ''' Writes values to a log file and while also printing to stdout. All output messages
    in `toast` should use this function to maintain consistency.

    This is a convenience function for succinct logging and user update messages.
    It is best practice to log all progress in a file so that it is available for later
    reference if questions arise or other users want to inquire about code execution.

    Args:
        message (str)               The message string to be logged and printed.
        file_path (str, optional):  Path to the log file for logging activity. Must include extension, if desired.
    
    Returns:
        No return value. Message is logged and printed to stdout.

    '''
    with open( file_path, 'a') as f:
        f.write('\n' + ttime.get_timestamp() + '  ' + str(message))
    print( message)
