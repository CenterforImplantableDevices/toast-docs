import matplotlib.pyplot as plt


[docs]def close_all_figures():
    '''Convenience function which clears and closes all existing figure instances
    created by matplotlib.pyplot.

    Closing all figure instances in helpful when attempting to control memory usage,
    as plot figures generate by matplotlib can often occupy a lot of space in memory
    and are sometimes difficult or annoying to track and remember to close.

    Args:
        None.

    Returns:
        No return value. All matplotlib.pyplot plots are individually cleared and closed.
    '''
    for fig in plt.get_fignums():
        plt.figure( fig)
        plt.clf()
        plt.close()
