import sys

[docs]def check_exit():
    '''Allows a user to kill a script gracefully. 
    If a file named ``exit`` is found in the current directory, ``sys.exit()`` will be called.

    Args:
        None

    Returns:
        None. Exits the current process.
    '''
    for filename in os.listdir('./'):
        if filename == 'exit':
            print('\n\nFound Exit File! Exiting...\n\n')
            sys.exit()





[docs]def check_finish():
    '''Allows a user to indicate a desired action from outside execution through the file system, 
    usually to finish and save current progress.
    
    If a file named ``finish`` is found in the current directory, True will be returned to indicate the users intervention.
    
    Args:
        None

    Returns:
        Boolean indicating whether to finish or not.
    '''
    for filename in os.listdir('./'):
        if filename == 'finish':
            print('\n\nFound Finish File! Finishing...\n\n')
            return True
    return False
