[docs]def track_peak_frequency( data, fs, min_freq, max_freq, window=10, step=2, zero_extend=6, interpolate_fcn=scipy.interpolate.CubicSpline):
    '''Tracks the peak frequency within a certain frequency range.
    ex. Track heart rate, which exists within 0.66 Hz - 4 Hz in humans (40 BMP - 240 BPM)
    
    Function can only handle one channel at a time.
    
    Note: The interpolation fuction used can be easily changed. Currently a cubic spline interpolation function is used.
        
    Note: An stft reduces the size of the data slightly.  To account for this the first and last
    few data points are extended from the first and last calculated values so that the
    size of the data matches the input size.

    Args:  
        data (numpy array):              Numpy array or ndarray (single row) representing the data to be filtered.
        fs (int):                        Sample rate of the data.
        min_freq (int or float):         The minimum expected frequency of the tracked frequency.
        max_freq (int or float):         The maximum expected frequency of the tracked frequency.
        window (int, optional):          Size of the window used when computing the fft for the stft used to track freq.
        step (int, optional):            Step size to use when sliding widow across data.
        zero_extend (int, optional):     Multiple of len(data) to zero extend by.  i.e. passing 2 would add 2*len(data) zeros to the end of each window of data.
        interpolate_fcn (fcn, optional): Function pointer to be used when inerpolating output data. Do not include parenthesis when passing function pointers.

    Returns:
        Numpy array or ndarray (single row) (matching the format passed) of peak frequency for the entire duration of data.
    '''

    # Windows and fft calculations are implimented using an stft with a square window (all ones) of length `window`
    #   Note this implimentation is kept separate from toast.butter.freq.fourier.stft implementation so that potential
    #   future changes in the windowing function used by stft() does not affect the accuracy of this function.
    window *= fs
    step   *= fs
    window_arr = np.ones(window)
    overlap    = int( (window - step) / 2)
    zero_extend = window * zero_extend

    # Padding no longer performed due to weird edge effects.
    # data = np.pad( data, int(window/2), mode='symmetric')

    freq_values, _, fft_values = signal.stft(data, fs=fs, window=window_arr, nperseg=window, noverlap=overlap, boundary=None, padded=False) # nfft=zero_extend,
    fft_values  = np.square( np.abs( fft_values))
    min_index   = np.argwhere((freq_values > min_freq) & (freq_values > 0))[0][0]
    max_index   = np.argwhere((freq_values < max_freq) & (freq_values > 0))[-1][0]
    
    freq_index = []
    for fft_window in fft_values.transpose():
        peak_index, peak_dict = signal.find_peaks( fft_window[ min_index:max_index], height=(None, None)) # Pass (None, None) to get it to calculate and return values
        try:
            freq_index.append( peak_index[ np.argmax( peak_dict['peak_heights'])] + min_index)
        except:
            freq_index.append( freq_index[-1])
    
    peak_freq = freq_values[ freq_index]

    new_x = np.linspace( 0, len(peak_freq)- 1, len(data)-window)
    # Use any funciton from the scipy.interpolate Univarite Interpolation functions: https://docs.scipy.org/doc/scipy/reference/interpolate.html
    # peak_freq = interpolate.interp1d(list(range(len(peak_freq))), peak_freq, kind='slinear')(new_x)
    # peak_freq = interpolate.CubicSpline(list(range(len(peak_freq))), peak_freq)(new_x)
    peak_freq = interpolate_fcn(list(range(len(peak_freq))), peak_freq)(new_x)

    # Extend dataframe to original length
    peak_freq_before = peak_freq[0]  * np.ones(int(window/2))
    peak_freq_after  = peak_freq[-1] * np.ones(int(window/2))


    return np.concatenate((peak_freq_before, peak_freq, peak_freq_after))




[docs]def track_mean_frequency( data, fs, min_freq, max_freq, window=10, step=2, zero_extend=6, interpolate_fcn=scipy.interpolate.CubicSpline):
    '''Tracks the centroid frequency within a certain frequency range.

    For example, can be used to track EEG frequency power. The mean frequency should rise as seizure activity begins.

    Note: The stft call in this function reduces the size of the data slightly.
          To account for this the first and last few data points are extended from the 
          first and last calculated values so that the size of the data matches the input size.

    Args:  
        data (numpy array or ndarray): Data to be filtered. **Function can only handle one channel at a time.**
        fs (int):                      Sample rate of the data.
        min_freq (float):              Minimum expected frequency of the tracked frequency.
        max_freq (float):              Maximum expected frequency of the tracked frequency.
        window (int, optional):        Size of the window used when computing the fft for the stft used to track freq.
        step (int, optional):          Step size to use when sliding widow across data.
        zero_extend (int, optional):   Multiple of len(data) to zero extend by.  i.e. passing 2 would add 2*len(data) zeros to the end of each window of data.
        interpolate_fcn (fcn, optional): Function pointer to be used when inerpolating output data. Do not include parenthesis when passing function pointers.

    Returns:
        Numpy array or ndarray (single row) (matching the format passed) of peak frequency for the entire duration of data.
    '''

    # Windows and fft calculations are implimented using an stft with a square window (all ones) of length `window`
    #   Note this implimentation is kept separate from toast.butter.freq.fourier.stft implementation so that potential
    #   future changes in the windowing function used by stft() does not affect the accuracy of this function.
    window *= fs
    step   *= fs
    window_arr = np.ones(window)
    overlap    = int( (window - step) / 2)
    zero_extend = window * zero_extend

    # Padding no longer performed due to weird edge effects.
    # data = np.pad( data, int(window/2), mode='symmetric')

    freq_values, _, fft_values = signal.stft(data, fs=fs, window=window_arr, nperseg=window, noverlap=overlap, boundary=None, padded=False) # nfft=zero_extend,
    fft_values  = np.square( np.abs( fft_values))
    min_index   = np.argwhere((freq_values > min_freq) & (freq_values > 0))[0][0]
    max_index   = np.argwhere((freq_values < max_freq) & (freq_values > 0))[-1][0]
    
    avg_freq = []
    for fft_window in fft_values.transpose():
        total_power = np.sum( fft_window[ min_index:max_index])
        freq_contributions = np.multiply( freq_values[min_index:max_index], fft_window[min_index:max_index])
        centroid = np.sum( np.divide( freq_contributions, total_power))
        avg_freq.append( centroid)

    new_x = np.linspace( 0, len(avg_freq)- 1, len(data)-window)
    # Use any funciton from the scipy.interpolate Univarite Interpolation functions: https://docs.scipy.org/doc/scipy/reference/interpolate.html
    # avg_freq = interpolate.interp1d(list(range(len(avg_freq))), avg_freq, kind='slinear')(new_x)
    # avg_freq = interpolate.CubicSpline(list(range(len(avg_freq))), avg_freq)(new_x)
    avg_freq = interpolate_fcn(list(range(len(avg_freq))), avg_freq)(new_x)

    # Extend dataframe to original length
    avg_freq_before = avg_freq[0]  * np.ones(int(window/2))
    avg_freq_after  = avg_freq[-1] * np.ones(int(window/2))


    return np.concatenate((avg_freq_before, avg_freq, avg_freq_after))

    '''Tracks the Upper Percentile frequency within a certain frequency range.
        
    For example, can be used to track EEG frequency power. The mean frequency should rise as seizure activity begins.

    Note: The stft call in this function reduces the size of the data slightly.
          To account for this the first and last few data points are extended from the 
          first and last calculated values so that the size of the data matches the input size.

    Args:  
        data (numpy array or ndarray):   Data to be filtered. **Function can only handle one channel at a time.**
        fs (int):                        Sample rate of the data.
        min_freq (float):                Minimum expected frequency of the tracked frequency.
        max_freq (float):                Maximum expected frequency of the tracked frequency.
        percentile (float, optional):    The frequency percentile to track.
        window (int, optional):          Size of the window used when computing the fft for the stft used to track freq.
        step (int, optional):            Step size to use when sliding widow across data.
        zero_extend (int, optional):     Multiple of len(data) to zero extend by.  i.e. passing 2 would add 2*len(data) zeros to the end of each window of data.
        interpolate_fcn (fcn, optional): Function pointer to be used when inerpolating output data. Do not include parenthesis when passing function pointers.

    Returns:
        Numpy array or ndarray (single row) (matching the format passed) of desired frequency percentile for the entire duration of data.
    '''

    # Windows and fft calculations are implimented using an stft with a square window (all ones) of length `window`
    #   Note this implimentation is kept separate from toast.butter.freq.fourier.stft implementation so that potential
    #   future changes in the windowing function used by stft() does not affect the accuracy of this function.
    window *= fs
    step   *= fs
    window_arr = np.ones(window)
    overlap    = int( (window - step) / 2)
    zero_extend = window * zero_extend

    # Padding no longer performed due to weird edge effects.
    # data = np.pad( data, int(window/2), mode='symmetric')

    freq_values, _, fft_values = signal.stft(data, fs=fs, window=window_arr, nperseg=window, noverlap=overlap, boundary=None, padded=False) # nfft=zero_extend,
    fft_values  = np.square( np.abs( fft_values))
    min_index   = np.argwhere((freq_values > min_freq) & (freq_values > 0))[0][0]
    max_index   = np.argwhere((freq_values < max_freq) & (freq_values > 0))[-1][0]
    
    avg_freq = []
    for fft_window in fft_values.transpose():
        avg_freq.append( freq_values[ min_index + np.argwhere( fft_window[min_index:max_index] == np.percentile( fft_window[min_index:max_index], percentile, interpolation='nearest'))][0][0])

    new_x = np.linspace( 0, len(avg_freq)- 1, len(data)-window)
    # Use any funciton from the scipy.interpolate Univarite Interpolation functions: https://docs.scipy.org/doc/scipy/reference/interpolate.html
    # avg_freq = interpolate.interp1d(list(range(len(avg_freq))), avg_freq, kind='slinear')(new_x)
    # avg_freq = interpolate.CubicSpline(list(range(len(avg_freq))), avg_freq)(new_x)
    avg_freq = interpolate_fcn(list(range(len(avg_freq))), avg_freq)(new_x)

    # Extend dataframe to original length
    avg_freq_before = avg_freq[0]  * np.ones(int(window/2))
    avg_freq_after  = avg_freq[-1] * np.ones(int(window/2))


    return np.concatenate((avg_freq_before, avg_freq, avg_freq_after))
