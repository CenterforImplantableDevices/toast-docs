[docs]def moving_rms( data, window, axis=-1, fast_approx=False):
    '''Calculates the Root Mean Square values from a window sliding along the requested axis.
    Window is cotinuously slid along the data, and is padded to match the shape of the input.

    Args:
        data (numpy array or ndarray): Data to process.
        window (int):                  Length of the window (number of points)
        axis (int, optional):          Array axis to slide the window accross.
        fast_approx (bool, optional):  Whether to approximate the rms value faster, by skipping the division and square root calculations.
    
    Returns:
        Numpy array matching the shape of the input array, containing the Root Mean Square values along the requested axis.
    '''
    calculated = moving_sum( np.square(data), window, axis=axis)
    if fast_approx:
        calculated = np.sqrt( np.divide( calculated, window))
    return calculated




[docs]def pad_to_length( data, desired_length, axis=-1):
    '''Helper function to pad data on both ends to match a desired length.

    The first and last values are used to add values before and after (respectively) the original data to meet the desired length.

    Args:
        data (numpy array):   The data to be padded
        desired_length (int): The length of the output data.
        axis (int, optional): Axis where padding should be added.
    
    Returns:
        Numpy array padded to match the desired length.
    '''
    pad_axes = [(0,0) for i in range(len(data.shape))]

    length_pad = desired_length - data.shape[-1]
    if length_pad % 2 == 0:
        pad_axes[axis] = (int(length_pad/2), int((length_pad)/2))
        return np.pad( data, pad_axes, mode='edge')
    else:
        pad_axes[axis] = (int(length_pad/2), int((length_pad+1)/2))
        return np.pad( data, pad_axes, mode='edge')




[docs]def moving_avg_cutoffHz( data, fs, cutoff, axis=-1):
    '''Calculates a moving window average filter. Window is cotinuously slid along the data,
    and is padded to match the shape of the input.

    The window sizes is estimated based on a requested frequency cutoff.

    Args:
        data (numpy array or ndarray): Data to process.
        fs (int):                      Sample Frequency of the data.
        cutoff (int):                  Desired frequency cutoff (Hz)
        axis (int, optional):          Array axis to slide the window accross.
    
    Returns:
        See :py:func:`~moving_avg`
    '''
    return moving_avg( data, int(((1/cutoff)/2)*fs), axis=axis)

    '''Calculates a moving window average filter, and the corresponding calculted standard deviation. 
    Window is cotinuously slid along the data, and is padded to match the shape of the input.

    The window sizes is estimated based on a requested frequency cutoff.

        axis (int, optional):          Array axis to slide the window accross.

        See :py:func:`~moving_avg`

[docs]def moving_1d_crossings( data, window, axis=-1):
    '''Calculates the number of first derivative zero crossings within a requested window length.
    The requested window is slid accross the data, and the zero crossings counted.

    Args:
        data (numpy array or ndarray): Data to process.
        window (int):                  Length of the window (number of points)
        axis (int, optional):          Array axis to slide the window accross.
        
    Returns:
        Numpy array matching the shape of the input array, containing the number of first derivative zero crossings along the requested axis.
    '''
    # count zero crosssings along a moving window accross first derivative
    first_derivative_sign_change = np.diff( np.signbit( np.diff( data))).astype(np.bool)

    pad_axes = [(0,) for i in range(len(data.shape))]
    pad_axes[axis] = (1,)
    return moving_sum( np.pad( first_derivative_sign_change, pad_axes, mode='constant', constant_values=0), window, axis=axis)

    The requested window is slid accross the data, and the zero crossings counted.

    Args:
        data (numpy array or ndarray): Data to process.
        window (int):                  Length of the window (number of points)
        axis (int, optional):          Array axis to slide the window accross.
        
    Returns:
        Numpy array matching the shape of the input array, containing the number of second derivative zero crossings along the requested axis.
    '''
    # count zero crosssings along a moving widow accross second derivative
    second_derivative_sign_change = np.diff( np.signbit( np.diff( np.diff( data)))).astype(np.bool)

    pad_axes = [(0,0) for i in range(len(data.shape))]
    pad_axes[axis] = (1,2)
    return moving_sum( np.pad( second_derivative_sign_change, pad_axes, mode='constant', constant_values=0), window, axis=axis)




[docs]def track_derivatives( data, fs, lpf_cutoff, window, axis=-1):
    '''Calculates the number of first and second derivative zero crossings within a requested window length.
    The requested window is slid accross the data, and the zero crossings counted.

    Data is filtered before derivatives are calculated. A butterworth low pass filter with the requested cutoff is applied.
    For more detail, see :py:func:`~toast.butter.filter.butterworth`

    Args:
        data (numpy array or ndarray): Data to process.
        fs (int):                      Sample Frequency of the data.
        lpf_cutoff (int):              Desired low pass filter frequency cutoff (Hz)
        window (int)                   Length of the window (number of points)
        axis (int, optional):          Array axis to slide the window accross.
        
    Returns:
        Two numpy arrays matching the shape of the input array, containing the number of first and second derivative zero crossings along the requested axis.
    '''
    if lpf_cutoff > 0:
        data_filtered = butterworth.lowpass_filter( data, fs, lpf_cutoff)
    else:
        data_filtered = data
    first_derivative  = moving_1d_crossings( data_filtered, window, axis=axis)
    second_derivative = moving_2d_crossings( data_filtered, window, axis=axis)
    return first_derivative, second_derivative




[docs]def track_number_exterma( data, max_index, min_index, window):
    '''Calculates the number of extrema (local max and local min) within a requested window length.
    The requested window is slid accross the data, and the zero crossings counted.

    Important:
        Function only works along a single axis

    Args:
        data (numpy array or ndarray): Data to process.
        max_index (iterable):          Indicies of identified local maximums.
        min_index (iterable):          Indicies of identified local minimums.
        window (int)                   Length of the window (number of points)
        
    Returns:
        Numpy array matching the shape of the input array, containing the number of extrema along the requested axis.
    '''
    peak_indicies = np.concatenate( (max_index, min_index), axis=-1)
    peak_values   = np.concatenate( (np.ones( max_index.shape[-1], dtype=bool), np.zeros( min_index.shape[-1], dtype=bool)))

    peak_hold      = interp1d( data[peak_indicies], peak_values, kind='previous', assume_sorted=False)( data)
    zero_crossings = moving_sum(  np.diff( peak_hold).astype(np.bool), window)
    return pad_to_length( zero_crossings, data.shape[-1])




[docs]def _move_bottleneck_fcn( fcn_pointer, data, window, axis=-1):
    '''Helper funciton, mapping toast functions to functions from the `bottleneck <https://bottleneck.readthedocs.io/en/latest/index.html>`_ library

    Automatically padds output to be the same shape as the input.
    '''
    filtered_data = fcn_pointer( data, window, min_count=window, axis=axis)
    return pad_to_length( np.delete( filtered_data, np.arange(window-1), axis=axis), data.shape[axis], axis=axis)
