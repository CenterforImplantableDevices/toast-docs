        '''Intended to replicate the TiSc architecture, but using CNN layers with time-invariant convolution.

        Implementation and usage should exaclty match :py:func:`~toast.head.timescaleN.TiscMlpN` to allow for side-by-side comparison.

        If length_input is set to `None`, then the largest window size is assumed to be the length of the input.

        Args:
            layer_desc (2D iterable):            2D Iterable: sequential list contianing elemnt lists that describe each layers architecture. Specific indes values are defined in detail above.
            stride (int, optional):              Step size of each window. A each scale's stride will be set to `min(stride, window_length)` to avoid skipping samples.
            num_tisc_channels (int, optional):  How many repetitions or independent channels of Time Scale layers to apply to the data.
            length_input (int, optional):       Size of the data that will be input into the network. Necessary if the largest TiSc window size does not cover the entire data length, to aid in dimension calculation.
            dropout (1D iterable, optional):     1D Iterable of Boolean Values: Sequentially decide if the correlating layer in layer_desc iterable should be followed by a dropout layer. If value(s) don't exist they will be set to False.
            dropout_perc(float, optional):       The percentage dropout to implement in the dropotu layers.
            nonlinearity_tisc (torch activation, optional):  pytorch.nn activation fuction to use within the TiSc operation. Pass a reference to the class, not an instance of the class (don't use parenthesis).
            nonlinearity (torch activation, optional):       pytorch.nn activation fuction to use. Pass a reference to the class, not an instance of the class (don't use parenthesis).
            nonlinearity_final (torch activation, optional): pytorch.nn activation fuction to use after the final layer. Pass a reference to the class, not an instance of the class (don't use parenthesis).
            tisc_scale_multiplier (int, optional):           Exponential base designating how fast the window size increases.
            path_save_progress (str, optional):  Path to the parent directory where progress and tracking data should be saved.
            description (str, optional):         A description describing the contents/structure of the network for easy reference.
            comment (str, optional):             A comment about what this network is being used to do/test/investigate for later identification of tests run.

        Returns: 
            An instance of the ConvolutionalN class representing a deep neural network.
        '''
        
        # IMPORTANT: Make sure input_shape includes a channels dimension, likely a [1, ...] or [3, ...] dimension before data/image size
        self.decision_threshold  = 0.5
        self.one_hot_output = True if (layer_desc[-1] > 1) else False

        # Define the expected layer types, used for organizing potential debugging output.
        layer_type_dict = { 'ConvLayer': [nn.Conv1d, nn.Conv2d, nn.Conv3d], 
                            'PoolLayer': [nn.MaxPool1d, nn.MaxPool2d, nn.MaxPool3d, nn.AvgPool1d, nn.AvgPool2d, nn.AvgPool3d], 
                            'LinrLayer': [nn.Linear], 
                            'NonlinLayer': [nonlinearity, nonlinearity_final]
                          } # Dropout layer type not identified, will be categorized as "Other"
    # Call Network __init__() to initialize standard network variables and functions
        super(ConvTiscMlpN, self).__init__( layer_type_dict=layer_type_dict, path_save_progress=path_save_progress, description=description, description_short='cnntisc', comment=comment)

        self.min_window       = layer_desc[0][0]
        self.max_window       = layer_desc[0][1]
        self.scale_multiplier = tisc_scale_multiplier
        if not (self.scale_multiplier == 2):
            print('ERROR: Only scale_multiplier=2 is currently supported. Setting scale_multiplier to 2.')
            self.scale_multiplier = 2

        # Determine the minimum and miximum exponential relting the the minimum/maximum window sizes.
        # Window sizes should be an even multiple of scale_multiplier, if not the windows will be reduced/expanded.
        if not math.log( self.min_window, self.scale_multiplier).is_integer():
            print('WARNING: min_window (', self.min_window, ') was not an even multiple of scale_multiplier (', self.scale_multiplier, '). Reducing min_window to compensate')
            self.min_window = self.scale_multiplier ** int( math.log( self.min_window, self.scale_multiplier))
            print('WARNING: New value of min_window: ', self.min_window)
        if not math.log( self.max_window, self.scale_multiplier).is_integer():
            print('WARNING: max_window (', self.max_window, ') was not an even multiple of scale_multiplier (', self.scale_multiplier, '). Expanding max_window to compensate')
            self.max_window = self.scale_multiplier ** int( math.log( self.max_window, self.scale_multiplier)+1)
            print('WARNING: New value of max_window: ', self.max_window)
        self.min_window_exponential = int( math.log( self.min_window, self.scale_multiplier))
        self.max_window_exponential = int( math.log( self.max_window, self.scale_multiplier))

        # Define the linear layers
        tisc_channels = []
        for _ in range(num_tisc_channels):
            layers_temp = []
            # TiSc Layers
            tisc_temp = []
            for scale in range( self.min_window_exponential, self.max_window_exponential):
                window_size = tisc_scale_multiplier ** scale
                if stride is None:
                    tisc_temp.append( nn.Conv1d( 1, 1, window_size, stride=window_size, bias=True))
                else:
                    tisc_temp.append( nn.Conv1d( 1, 1, window_size, stride=min(stride,window_size), bias=True))
            layers_temp.append(nn.Sequential(*tisc_temp))
            # Nonlinearity
            if nonlinearity_tisc is not None:
                layers_temp.append(nonlinearity_tisc())
            # Dropout Layers
            layers_temp.append(nn.Dropout(p=dropout_perc))
            # Finalize layers as sequential object
            tisc_channels.append( nn.Sequential(*layers_temp))
        
        # Wrap all tisc channels in one sequential object
        tisc_layers    = nn.Sequential( *tisc_channels)

        # Define MLP part of network
        if stride is None:
            if length_input is None:
                mlp_layerSize = [ (layer_desc[0][1] * tisc_scale_multiplier - layer_desc[0][1] // self.max_window -2) * num_tisc_channels ]
            else:
                mlp_layerSize = [ ((length_input // self.min_window) * tisc_scale_multiplier - length_input // self.max_window -2) * num_tisc_channels ]
        else:
            window_sizes    = [tisc_scale_multiplier ** scale for scale in range( self.min_window_exponential, self.max_window_exponential)]
            tisc_outputSize = [((length_input-win) // min(stride,win)) +1 for win in window_sizes]
            mlp_layerSize   = [sum(tisc_outputSize) * num_tisc_channels]
        mlp_layerSize.extend(layer_desc[1:])
        mlp_layers    = toast_perceptron.define_sequential_linear_layers( mlp_layerSize, dropout=dropout, dropout_perc=dropout_perc, nonlinearity=nonlinearity, nonlinearity_final=nonlinearity_final)
        
        # Combine TiSc and MLP components in one sequential object
        self.layers   = nn.Sequential( tisc_layers, mlp_layers)
        self.new_layers_added()

        


[docs]    def forward( self, data):
        '''Overwrites the forward method of the Network class to process data based on our network structure.

        Args:
            data: Data to be processed by the network.

        output: The output of the network
        '''
        tisc_output_channels = []
        for channel in self.layers[0]:
            tisc_output_windows = []
            for tisc_window in channel[0]:
                temp = tisc_window( data)
                tisc_output_windows.append( temp)
            output_windows = torch.cat( tisc_output_windows, dim=-1)
            if len(channel) > 1:
                tisc_output_channels.append( channel[1:]( output_windows))
            else:
                tisc_output_channels.append( output_windows)
        tisc_output = torch.cat( tisc_output_channels, dim=-1)
            
        return self.layers[1]( tisc_output.reshape( tisc_output.shape[0], -1))

    # Add decoder layers together in order: ConvLayer -> dropout -> Upsampling  -> dropout -> repeat
    enc_layers     = []
    dec_blocks_rev = []
    layer_dimensions = [input_shape]
    channels_in = input_shape[0]

        if get_decode:
            dec_block = []
        keyword_args = {}
    # Convolutional Layer

                if   isinstance( layer[4], bool):

            # if len(layer) == 6:                         # TODO Add this to convolutionalN
            #     if isinstance(layer[5], bool):
            #         batchnorm[layer_num] = layer[5]
            #     elif isinstance(layer[5], dict):
            #         keyword_args = layer[5]
            # Determine the dimension of the convolution, and select the right class
            if   len(layer[1]) == 1:
                current_enc_layer_type  = nn.Conv1d
                current_dec_layer_type  = nn.ConvTranspose1d
                current_norm_layer_type = nn.BatchNorm1d
            elif len(layer[1]) == 2:
                current_enc_layer_type  = nn.Conv2d
                current_dec_layer_type  = nn.ConvTranspose2d
                current_norm_layer_type = nn.BatchNorm2d
            elif len(layer[1]) == 3:
                current_enc_layer_type  = nn.Conv3d
                current_dec_layer_type  = nn.ConvTranspose3d
                current_norm_layer_type = nn.BatchNorm3d
            # Convert stride to a list for dimension tracking
            if isinstance( layer[2], int):
                layer[2] = [layer[2]] * len(layer[1])

            # Add the convoltional layers, in order of: Convolution -> nonLinearity -> dropout
            enc_layers.append( current_enc_layer_type(channels_in, channels_in*layer[3], layer[1], stride=layer[2], **keyword_args))

            if get_decode:
                # Add deconvolution layer block (the order will be reversed after all the blocks have been added)
                output_padding = [ (layer_dimensions[-1][a] - layer[1][a]) % layer[2][a] for a in range(-1*len(layer[1]), 0) ]
                dec_block.append( current_dec_layer_type( channels_in * layer[3], channels_in, layer[1], stride=layer[2], output_padding=output_padding, **keyword_args))

            # TODO Add batchnorm
            # if batchnorm[layer_num]:
            #     enc_layers.append(current_norm_layer_type(channels_in * layer[3]))
            #     dec_block.append( current_norm_layer_type(channels_in))

            if nonlinearity is not None:
                enc_layers.append( nonlinearity())
                if get_decode and (layer_num > 0):
                    dec_block.append( nonlinearity())
            if dropout[layer_num]:
                enc_layers.append( nn.Dropout( p=dropout_perc))
                if get_decode:
                    dec_block.append(nn.Dropout(p=dropout_perc))
            channels_in = channels_in * layer[3]
            # Track the dimensions
            # from pytorch.org: dim_out = ((dim_in + 2*padding - dilation*(kernel_size-1) - 1) / stride) + 1 for each respective dim_in/padding/kilation/kernel/stride dimension
            layer_dimensions.append( [channels_in] + [int(((og_size - (kernel_size-1) - 1) / stride) + 1) for og_size, kernel_size, stride in zip( layer_dimensions[-1][1:], layer[1], layer[2])])
                                                                                                                                                                      # ^^ to skip the channels dim
    # MaxPool Layer
        elif layer[0].lower() == 'max':
            # If additional specs exist
            if len(layer) == 4:
                if   isinstance( layer[3], bool):
                    dropout[layer_num] = layer[3]
                elif isinstance( layer[3], dict):
                    keyword_args = layer[3]
            # if len(layer) == 5:                         # TODO Add this to convolutionalN
            #     if isinstance(layer[4], bool):
            #         batchnorm[layer_num] = layer[4]
            #     elif isinstance(layer[4], dict):
            #         keyword_args = layer[4]
            # Determine the dimension of the MaxPool, and select the right class
            if   len(layer[1]) == 1:
                current_enc_layer_type  = nn.MaxPool1d
                current_norm_layer_type = nn.BatchNorm1d
            elif len(layer[1]) == 2:
                current_enc_layer_type  = nn.MaxPool2d
                current_norm_layer_type = nn.BatchNorm2d
            elif len(layer[1]) == 3:
                current_enc_layer_type  = nn.MaxPool3d
                current_norm_layer_type = nn.BatchNorm3d
            current_dec_layer_type = nn.Upsample
            # Convert stride to a list for dimension tracking
            if isinstance( layer[2], int):
                layer[2] = [layer[2]] * len(layer[1])

            # Add the pool layers, in order of: MaxPool -> dropout
            enc_layers.append(current_enc_layer_type( layer[1], stride=layer[2], **keyword_args))
            if get_decode:
                dec_block.append( current_dec_layer_type( size=layer_dimensions[-1][1:]))
            # TODO Add Batchnorm
            # if batchnorm[layer_num]:
            #     enc_layers.append(current_norm_layer_type(channels_in))
            #     dec_block.append(current_norm_layer_type(channels_in))
            if dropout[layer_num]:
                enc_layers.append( nn.Dropout( p=dropout_perc))
                if get_decode:
                    dec_block.append(nn.Dropout(p=dropout_perc))
            # Track the dimensions
            # from pytorch.org: dim_out = ((dim_in + 2*padding - dilation*(kernel_size-1) - 1) / stride) + 1 for each respective dim_in/padding/kilation/kernel/stride dimension
            layer_dimensions.append( [channels_in] + [int(((og_size - (kernel_size-1) - 1) / stride) + 1) for og_size, kernel_size, stride in zip( layer_dimensions[-1][1:], layer[1], layer[2])])
                                                                                                                                                                      # ^^ to skip the channels dim
    # AvgPool Layer
        elif layer[0] == 'avg':
            # If additional specs exist
            if len(layer) == 4:
                if   isinstance( layer[3], bool):
                    dropout[layer_num] = layer[3]
                elif isinstance( layer[3], dict):
                    keyword_args = layer[3]
            if len(layer) == 5:                         # TODO Add this to convolutionalN
                if isinstance(layer[4], bool):
                    batchnorm[layer_num] = layer[4]
                elif isinstance(layer[4], dict):
                    keyword_args = layer[4]
            # Determine the dimension of the AvgPool, and select the right class
            if   len(layer[1]) == 1:
                current_enc_layer_type = nn.AvgPool1d
                current_norm_layer_type = nn.BatchNorm1d
            elif len(layer[1]) == 2:
                current_enc_layer_type = nn.AvgPool2d
                current_norm_layer_type = nn.BatchNorm2d
            elif len(layer[1]) == 3:
                current_enc_layer_type  = nn.AvgPool3d
                current_norm_layer_type = nn.BatchNorm3d
            current_dec_layer_type = nn.Upsample
            # Convert stride to a list for dimension tracking
            if isinstance( layer[2], int):
                layer[2] = [layer[2]] * len(layer[1])

            # Add the pool layers, in order of: AvgPool -> batchnorm -> dropout
            enc_layers.append( current_enc_layer_type( layer[1], stride=layer[2], **keyword_args))
            if get_decode:
                dec_block.append( current_dec_layer_type( size=layer_dimensions[-1][1:]))

            if batchnorm[layer_num]:
                enc_layers.append(current_norm_layer_type(channels_in))
                if get_decode:
                    dec_block.append( current_norm_layer_type(channels_in))

            if dropout[layer_num]:
                enc_layers.append( nn.Dropout( p=dropout_perc))
                if get_decode:
                    dec_block.append(nn.Dropout(p=dropout_perc))
            # Track the dimensions
            # from pytorch.org: dim_out = ((dim_in + 2*padding - kernel_size) / stride) + 1 for each respective dim_in/padding/kilation/kernel/stride dimension
            layer_dimensions.append( [channels_in] + [int(((og_size - kernel_size) / stride) + 1) for og_size, kernel_size, stride in zip( layer_dimensions[-1][1:], layer[1], layer[2])])
                                                                                                                                                              # ^^ to skip the channels dim
        else:
            print('\n\nWARNING: UNEXPECTED LAYER TYPE REQUESTED: ', layer, '\nSKIPPING FOR NOW...')
            continue

        if get_decode and (len(dec_block) > 0):
            dec_blocks_rev.append(dec_block)

    if get_decode:
        #reverse the decode blocks to get the correct decoder order and record layer dimensions.
        dec_layers = []
        index_layer_dims = len(layer_dimensions)
        for block in dec_blocks_rev[::-1]:
            dec_layers.extend(block)
            index_layer_dims -= 1

            if type(block[0]) in [nn.ConvTranspose1d, nn.ConvTranspose2d, nn.ConvTranspose3d]:
                channels_in = block[0].out_channels
                # From pytorch: dim_out = (dim_in − 1) * stride − 2 * padding + dilation * (kernel_size − 1) + output_padding + 1
                layer_dimensions.append([channels_in] + [(og_size - 1) * stride + (kernel_size - 1) + output_pad + 1 for og_size, stride, kernel_size, output_pad in zip( layer_dimensions[index_layer_dims][1:], block[0].stride, block[0].kernel_size, block[0].output_padding)] )
            elif type(block[0] is nn.Upsample):
                layer_dimensions.append([channels_in] + block[0].size) #Number of channels will be preserved, '' added for filler

        return nn.Sequential(*enc_layers), nn.Sequential(*dec_layers), layer_dimensions

    else:
        return nn.Sequential(*enc_layers), layer_dimensions
