[docs]    def __init__(self, path_data, path_sample_descriptions, label_key=None, needs_validation=True, one_hot_encode=False, data_files=None, data_type=npInt16):
        '''Initialization function to create an object instance of data.

        Args:  
            path_data (str):                   Path to the subject folder containing raw data. This joined the 'filename' feild of a sample dict should identify the raw data file.
            path_sample_descriptions (str):    Path to the .pkl file containing the sample dict that will be used for training. Note this is only the file prefix, do not include the "_train.pkl" part of the filename.
            label_key (dict, optional):        Dictionary that will be used to convert the "label_int" field of the sample dict to the "label_out" feild that will be used by the network during training.
            needs_validation (bool, optional): Whether a validation set is needed
            one_hot_encode (bool, optional):   Whether the output should be on_hot_encoded
            data_files (str, optional):        Suffix of data filenames designating which dataset organization to use. Necessary when following CID convention.
            data_type (np dtype, optional):    Data type to use when unpacking the data

        Returns:
            An instance of the Data_Bread class to be used to manage the dataset.
        '''
        self.path_raw_data = path_data
        self.path_sample_descriptions = path_sample_descriptions
        self.label_key = label_key
        self.data_files = data_files
        self.data_type = data_type

        self.samples_training = []
        self.has_validation = needs_validation
        if self.has_validation:
            self.samples_validation = []
        self.samples_testing = []
        
        self.filenames_sample_descriptions = []
        self.samples_all_list = []

        self.one_hot_key = None
        self.training    = None
        self.testing     = None
        self.validation  = None

        # training/testing/validation attributes must be defined differently based on if a validation is included.
        # Different .pkl files are imported, and the attributes themselves must be defined appropriately, requiring 2 separate intialization flows.
        if needs_validation:
            training_perc   = 0.50
            validation_perc = 0.25
            testing_perc    = 0.25
        else:
            training_perc   = 0.70
            testing_perc    = 0.30

        # Append the appropriate ending to the targeted filenames.
        self.filenames_sample_descriptions.append( join( splitext(self.path_sample_descriptions)[0] + '_train-' + str(int(training_perc*100)) + '.pkl'))
        self.filenames_sample_descriptions.append( join( splitext(self.path_sample_descriptions)[0] + '_test-' + str(int(testing_perc*100)) + '.pkl'))
        if needs_validation:
            self.filenames_sample_descriptions.append( join( splitext(self.path_sample_descriptions)[0] + '_val-' + str(int(validation_perc*100)) + '.pkl'))

        # If any of these sample files don't already exist, we must generate them
        if any( [not( exists( p)) for p in self.filenames_sample_descriptions]):
            with open(self.path_sample_descriptions, 'rb') as f:
                samples = pickle.load(f)

            for sample_for_single_label in samples:
                random.shuffle( sample_for_single_label)
                if needs_validation:
                    self.samples_training.extend(   sample_for_single_label[ : int(training_perc*len(sample_for_single_label))])
                    self.samples_validation.extend( sample_for_single_label[int(training_perc*len(sample_for_single_label)) : int((training_perc+validation_perc)*len(sample_for_single_label))])
                    self.samples_testing.extend(    sample_for_single_label[int((training_perc+validation_perc)*len(sample_for_single_label)) : ])
                else:
                    self.samples_training.extend( sample_for_single_label[ : int(training_perc*len(sample_for_single_label))])
                    self.samples_testing.extend(  sample_for_single_label[int(training_perc*len(sample_for_single_label)) : ])

        # Otherwise just load the existing files
        else:
            with open( self.filenames_sample_descriptions[0], 'rb') as f:
                self.samples_training = pickle.load(f)
            with open( self.filenames_sample_descriptions[1], 'rb') as f:
                self.samples_testing = pickle.load(f)
            if needs_validation:
                with open( self.filenames_sample_descriptions[2], 'rb') as f:
                    self.samples_validation = pickle.load(f)

        # Append all of the samples into one list
        self.samples_all_list.append( self.samples_training)
        self.samples_all_list.append( self.samples_testing)
        if needs_validation:
            self.samples_all_list.append( self.samples_validation)

        # Define a default label key, if one is not passed
        if self.label_key is None:
            label_int_unique = []
            for sample_set in self.samples_all_list:
                for s in sample_set:
                    if s['label_int'] not in label_int_unique:
                        label_int_unique.append( s['label_int'])
            self.label_key = {}
            [self.label_key.update({i: i}) for i in label_int_unique]

        # One hot encode the samples if necessary
        if one_hot_encode:
            # Note: set will remove duplicate values
            self.one_hot_key = sorted( set( self.label_key.values()))
        
        self.set_sample_labels(one_hot_encode=one_hot_encode)

        # Save these results to a file, so they don't need to be calculated every time
        for dataset_filename, data_samples in zip( self.filenames_sample_descriptions, self.samples_all_list):
            with open( dataset_filename, 'wb') as f:
                pickle.dump( data_samples, f)

        self.training = Dataset_Bread( self.path_raw_data, self.samples_training, self.label_key, data_files=self.data_files, data_type=self.data_type)
        self.testing  = Dataset_Bread( self.path_raw_data, self.samples_testing,  self.label_key, data_files=self.data_files, data_type=self.data_type)
        if needs_validation:
            self.validation = Dataset_Bread( self.path_raw_data, self.samples_validation, self.label_key, data_files=self.data_files, data_type=self.data_type)



[docs]    def set_sample_labels(self, one_hot_encode=False):
        ''' Define label_out, which will be used in `__get_item__()` in the Dataset_Bread object definition

            Also find the relative occurance of each label, and define a weighting to use when calculating loss.

            Args:
                one_hot_encode (bool, optional): Whether to one_hot_encode the labels
        '''
        self.num_unique_labels = len(self.label_key)
        label_counts = [0] * self.num_unique_labels
        for sample_set in self.samples_all_list:
            for s in sample_set:
                s_label = self.label_key[ s['label_int']]
                # Generate network output labels
                if one_hot_encode:
                    s['label_out'] = self.one_hot_encode( s_label, num_labels=self.num_unique_labels)
                else:
                    s['label_out'] = [s_label]
                # Track label counts
                if one_hot_encode:
                    label_counts[ self.one_hot_key.index( s_label)] += 1
                else:
                    label_counts[ s_label] += 1
        if 0 in label_counts:
            print('WARNING: At least one indicated label has no samples. Please include at least one sample of each label.')
            self.loss_weight = None
        else:
            min_samples = min(label_counts)
            self.loss_weight = [ min_samples / cnt for cnt in label_counts]

        '''Used to generate k folds in the training dataset to allow for k-fold training.

        Args:
            number_folds (int):              How many folds to divide the training set into.
            plot_foldGroup (bool, optional): Whether to display a plot visualizing the fold groups. Typically used for visual confirmation of data splits.

        Returns:
            No return value. Generates the samples_training_fold_traingroup attribute 
            containing a 2D list: 1D-data for each training execution on the respective 
            fold iteration/index, 2D-contains the [training_dataset, testing_dataset] for 
            each fold iteration.
        '''
        length_fold = int( len( self.samples_training) / number_folds)
        
        if plot_foldGroup:
            import pandas as pd
            data_all = pd.DataFrame(self.samples_training + self.samples_testing)
        
        # Determine the folds
        self.samples_training_fold = []
        for i in range( number_folds):
            self.samples_training_fold.append( self.samples_training[ i*length_fold : (i+1)*length_fold])

        # Use the folds to create a list of training/testing dataset pairs
        self.samples_training_fold_traingroup = []
        for testing_i in range( number_folds):
            training_samples = []
            testing_samples  = []
            for j in range( number_folds):
                if j == testing_i:
                    testing_samples.extend(  self.samples_training_fold[j])
                else:
                    training_samples.extend( self.samples_training_fold[j])

            if plot_foldGroup:
                data_all['group'] = 0
                print('...labeling training...')
                for s in training_samples:
                    s_row = data_all[ data_all['start_datetime'] == s['start_datetime']]
                    if s_row.shape[0] > 1:
                        print('ERROR DUPLICATE SAMPLE')
                        print(s_row)
                        print(s_row.shape)
                    data_all.loc[ data_all['start_datetime'] == s['start_datetime'], 'group'] += 1
                print('...labeling testing...')
                for s in testing_samples:
                    s_row = data_all[ data_all['start_datetime'] == s['start_datetime']]
                    if s_row.shape[0] > 1:
                        print('ERROR DUPLICATE SAMPLE')
                    data_all.loc[ data_all['start_datetime'] == s['start_datetime'], 'group'] += 3
                for s in self.samples_testing:
                    s_row = data_all[ data_all['start_datetime'] == s['start_datetime']]
                    if s_row.shape[0] > 1:
                        print('ERROR DUPLICATE SAMPLE')
                    data_all.loc[ data_all['start_datetime'] == s['start_datetime'], 'group'] += 7
                
                import matplotlib.pyplot as plt
                plt.plot(data_all['start_datetime'], data_all['group'], '.')
                plt.title('Fold Training Group '+str(testing_i))
                plt.ylabel('Category (1=train, 2=test)')
                plt.show()
            
            training_dataset = Dataset_Bread( self.path_raw_data, training_samples, self.label_key, data_files=self.data_files, data_type=self.data_type)
            testing_dataset  = Dataset_Bread( self.path_raw_data, testing_samples, self.label_key, data_files=self.data_files, data_type=self.data_type)
            self.samples_training_fold_traingroup.append( [training_dataset, testing_dataset])

[docs]    def one_hot_encode( self, label, num_labels=None):
        '''Returns the corresponding one_hot_encoding for a given label.

        Args:  
            label (int):                Integer value of the desired label
            num_labels (int, optional): How many unique labels exist, i.e. how many possible one-hot values to include.

        Returns:
            encoding (list):                The one_hot encoded version of the input label integer.
        '''
        if num_labels is None:
            num_labels = self.num_unique_labels
        encoding = [0] * num_labels
        encoding[ self.one_hot_key.index( label)] = 1
        return encoding

        Args:
            data_recall_return (numpy array): First index of values returned directly from `toast bread.load.*.recall_data*`, 
                                              containing the numpy data to process. For more information, see :py:func:`~toast.bread.load.wav.recall_data`
            sample_metadata (int):            Second index of values returned directly from `toast bread.load.*.recall_data*`, 
                                              containing the number of channels. For more information, see :py:func:`~toast.bread.load.wav.recall_data`

        Returns:
            data_as_numpy (numpy array): The data array after any desired preprocessing functions are applied.

        '''
        # Note: You must return a copy of the ndarray so that it is writable, as PyTorch does not support non-writable tensors
        return data_recall_return[0].copy()


    def __len__(self):

    def __getitem__(self, idx):
        sample = self.samples[idx]
        try:
            if not self.show_timings:
                return torch.from_numpy( self.preprocess( wav.recall_data( self.path_raw_data, sample['filename'], sample['start_index'], sample['stop_index'], dataset=self.data_files, datatype=self.data_type), sample).astype(npFloat32)), torch.tensor( sample['label_out']).float()
            else:
                self.show_timings_count += 1
                if self.show_timings_count > 100:
                    self.show_timings = False
                    self.show_timings_count = 0
                start_time = time.time()
                data = wav.recall_data( self.path_raw_data, sample['filename'], sample['start_index'], sample['stop_index'], dataset=self.data_files)
                print('\t\t\tTime to read data from disk:\t{:>9.6f} s'.format(time.time() - start_time))
                start_time = time.time()
                data = self.preprocess( data, sample)
                print('\t\t\tTime to preprocess data:\t{:>9.6f} s'.format(time.time() - start_time))
                start_time = time.time()
                data = data.astype(npFloat32)
                print('\t\t\tTime to npcast to float:\t{:>9.6f} s'.format(time.time() - start_time))
                start_time = time.time()
                data = torch.from_numpy( data)
                print('\t\t\tTime to cast to tensor:\t\t{:>9.6f} s'.format(time.time() - start_time))
                # start_time = time.time()
                # data = data.float()
                # print('\t\t\tTime to cast to float:\t\t{:>9.6f} s'.format(time.time() - start_time))
                return data, torch.tensor( sample['label_out']).float()
                
        except Exception as e:
            print('ERROR reading sample ', str(idx), ' from ', sample['filename'])
            print(e)
            return -1
