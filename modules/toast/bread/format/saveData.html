    '''Wrapper function to allow one common-use function for multiple file types

    Much of the functionality of this function is determined by the funcitons below. Reference those relevant functions for execution details.

    Args:
        raw_data (dict):        Dictionary of numpy arrays 
        label_group (list):     Python list of strings (keys in raw_data) identifying the groups for the "_label" file
        data_group (list):      Python list of strings (keys in raw_data) identifying the groups for the "_data" file
        metadata_group (list):  Python list of strings (keys in raw_data) identifying the groups for the "_metadata" file
        folder_path (str):      File path where the raw data files are saved.
        filename (str):         Filename of the raw data file (used as filename prefix for newly formatted data)
        save_path (str):        File path where the output files should be saved.
        sample_frequency (int): frequency that the data is sampled at.
        out (str, optional):    Which file format to save with. `.wav` is the standard for CID applications.
        source (str, optional): Where the source of the data is. This mechanism helps account for varrying file/folder structures in different data sources.
        dataset_suffix (str, optional):  Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        path_log (str, optional):        Path to the file for outputing logs during execution.
            
    Returns:
        No return value. Generates the "_label" "_data" and "_metadata" files in the save_path location.
    '''
    if out == 'wav':
        save_to_wav( save_path, filename, raw_data, label_group, data_group, metadata_group, sample_frequency, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'hdf5':
        if source == 'synology':
            file_path_bin  = os.path.join( folder_path, 'Original Bin Files', filename + ".bin")
        elif source == 'test':
            file_path_bin  = os.path.join( folder_path, filename + ".bin")
        save_to_hdf5( save_path, filename, raw_data, label_group, data_group, metadata_group, file_path_bin, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'hdf5c':
        if source == 'synology':
            file_path_bin   = os.path.join( folder_path, filename + ".bin")
        elif source == 'test':
            file_path_bin   = os.path.join( folder_path, filename + ".bin")
        save_to_hdf5c( save_path, filename, raw_data, label_group, data_group, metadata_group, file_path_bin, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'hdf5_pandas':
        save_to_hdf5_pandas( save_path, filename, raw_data, label_group, data_group, metadata_group, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'pkl':
        save_to_pkl( save_path, filename, raw_data, label_group, data_group, metadata_group, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'edf':
        save_to_edf( save_path, filename, raw_data, label_group, data_group, metadata_group, dataset_suffix=dataset_suffix, path_log=path_log)
    elif out == 'csv':
        save_to_csv( save_path, filename, raw_data, label_group, data_group, metadata_group, dataset_suffix=dataset_suffix, path_log=path_log)
    else:
        tlog.output_log('INVALID OUTPUT TYPE SPECIFIED. OUTPUT FILE WILL NOT BE SAVED', path_log)





[docs]def create_dataframe( data, label_group, data_group, metadata_group, path_log='./log.txt'):
    '''Convert a list of dictionaries to a dataframe for better structure for saving.

    Args:
        data (dict):              Dictionary of numpy arrays 
        label_group (list):       Python list of strings (keys in raw_data) identifying the groups for the "_label" file
        data_group (list):        Python list of strings (keys in raw_data) identifying the groups for the "_data" file
        metadata_group (list):    Python list of strings (keys in raw_data) identifying the groups for the "_metadata" file
        path_log (str, optional): Path to the file for outputing logs during execution.
                        
    Returns:
        df_label, df_data, df_metadata: Three dataframes representing the label_group, data_group, and metadata_group data (respectively)
    '''
    import pandas as pd
    tlog.output_log('Converting Data to DataFrame', path_log)

    df_label = pd.DataFrame()
    for data_name in label_group:
        df_label[data_name] = data[data_name]

    df_data = pd.DataFrame()
    for data_name in data_group:
        df_data[data_name] = data[data_name]

    df_metadata = pd.DataFrame()
    for data_name in metadata_group:
        df_metadata[data_name] = data[data_name]

    return df_label, df_data, df_metadata





[docs]def save_to_wav( save_path, filename, data_dict, label_group, data_group, metadata_group, sample_frequency, dataset_suffix=None, path_log='./log.txt'):
    '''Saves raw-data, label-data, and meta-data to separate `.wav` files.

    These files are saved in a standardized way to assure compatibility with other toast packages/modules.

    Args:
        save_path (str):        Path where the data should be saved
        filename (str):         Filename prefix to use when generating the data files.
        data_dict (dict):       Dictionary of numpy arrays 
        label_group (list):     Python list of strings (keys in raw_data) identifying the groups for the "_label" file
        data_group (list):      Python list of strings (keys in raw_data) identifying the groups for the "_data" file
        metadata_group (list):  Python list of strings (keys in raw_data) identifying the groups for the "_metadata" file
        sample_frequency (int): Frequency that the data is sampled at.
        dataset_suffix (str, optional):  Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        path_log (str, optional):        Path to the file for outputing logs during execution.
    
    Returns:
        No return value. Generates the "_label" "_data" and "_metadata" files in the save_path locaiton.

    '''
    import numpy as np
    import wave

    if dataset_suffix is None:
        dataset_suffix = ''
    else:
        dataset_suffix = '-' + dataset_suffix

    filename_label = os.path.join(save_path, filename+'_label'+dataset_suffix+'.wav')
    filename_data  = os.path.join(save_path, filename+'_data'+dataset_suffix+'.wav')
    filename_meta  = os.path.join(save_path, filename+'_metadata'+dataset_suffix+'.wav')

    if len(label_group) > 0:
        tlog.output_log( 'Saving File: '+filename_label, path_log)
        with wave.open( filename_label, 'w') as f:
            f.setparams((len(label_group), data_dict[label_group[0]].itemsize, sample_frequency, 0, 'NONE','not compressed'))
            array_to_write = []
            for dataset_name in label_group:
                array_to_write.append( data_dict[dataset_name])
            
            np_array_to_write = np.asarray( [val for frame in zip(*array_to_write) for val in frame])
            f.writeframes( np_array_to_write)
        
    if len(data_group) > 0:
        tlog.output_log( 'Saving File: '+filename_data, path_log)
        with wave.open( filename_data, 'w') as f:
            f.setparams((len(data_group), data_dict[data_group[0]].itemsize, sample_frequency, 0, 'NONE','not compressed'))
            array_to_write = []
            for dataset_name in data_group:
                array_to_write.append( [data_dict[dataset_name]])
            
            np_array_to_write = np.dstack(tuple(array_to_write)).flatten()
            f.writeframes( np_array_to_write)
        
    if len(metadata_group) > 0:
        tlog.output_log( 'Saving File: '+filename_meta, path_log)
        with wave.open( filename_meta, 'w') as f:
            f.setparams((len(metadata_group), data_dict[metadata_group[0]].itemsize, sample_frequency, 0, 'NONE','not compressed'))
            array_to_write = []
            for dataset_name in metadata_group:
                array_to_write.append( data_dict[dataset_name])
            
            np_array_to_write = np.asarray( [val for frame in zip(*array_to_write) for val in frame])
            f.writeframes( np_array_to_write)





[docs]def save_to_hdf5( save_path, filename, data_dict, label_group, data_group, metadata_group, file_path_bin=None, dataset_suffix=None, path_log='./log.txt'):
    '''Saves raw-data, label-data, and meta-data to separate `.hdf5` files.

    These files are saved in a standardized way to assure compatibility with other toast packages/modules.

    Uses the h5py library.

    Args:
        save_path (str):       Path where the data should be saved
        filename (str):        Filename prefix to use when generating the data files.
        data_dict (dict):      Dictionary of numpy arrays 
        label_group (list):    Python list of strings (keys in raw_data) identifying the groups for the "_label" file
        data_group (list):     Python list of strings (keys in raw_data) identifying the groups for the "_data" file
        metadata_group (list): Python list of strings (keys in raw_data) identifying the groups for the "_metadata" file
        file_path_bin (str, optional):   Path to the original .bin file. Used to read the header line and transfer to hdf5 file.
        dataset_suffix (str, optional):  Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        path_log (str, optional):        Path to the file for outputing logs during execution.
    
    Returns:
        No return value. Generates the "_label" "_data" and "_metadata" files in the save_path locaiton.

    '''
    import h5py

    if dataset_suffix is None:
        dataset_suffix = ''
    else:
        dataset_suffix = '-' + dataset_suffix

    filename_out = os.path.join(save_path, filename+'.hdf5')
    tlog.output_log( 'Saving File: '+filename_out, path_log)
    with h5py.File( filename_out, 'w', userblock_size=512, libver='latest') as file_hdf5:

        if len(label_group) > 0:
            hdf5_labels = file_hdf5.create_group("label"+dataset_suffix)
            for dataset_name in label_group:
                hdf5_labels.create_dataset(dataset_name, shape=data_dict[dataset_name].shape, dtype=data_dict[dataset_name].dtype, data=data_dict[dataset_name] , chunks=True)

        if len(data_group) > 0:
            hdf5_data = file_hdf5.create_group("data"+dataset_suffix)
            for dataset_name in data_group:
                hdf5_data.create_dataset(dataset_name, shape=data_dict[dataset_name].shape, dtype=data_dict[dataset_name].dtype, data=data_dict[dataset_name] , chunks=True)

        if len(metadata_group) > 0:
            hdf5_metadata = file_hdf5.create_group("metadata"+dataset_suffix)
            for dataset_name in metadata_group:
                hdf5_metadata.create_dataset(dataset_name, shape=data_dict[dataset_name].shape, dtype=data_dict[dataset_name].dtype, data=data_dict[dataset_name] , chunks=True)

    if (file_path_bin is not None) and (os.path.exists(file_path_bin)):
        with open(file_path_bin, 'rb+') as file_hdf5:
            with open(file_path_bin, 'rb') as f:
                header_line = f.read(512)
            file_hdf5.write(header_line)

            for dataset_name in data_group:
                hdf5_data.create_dataset(dataset_name, shape=data_dict[dataset_name].shape, dtype=data_dict[dataset_name].dtype, data=data_dict[dataset_name], compression='gzip', compression_opts=compression_level, chunks=True)

        if len(metadata_group) > 0:
            hdf5_metadata = file_hdf5.create_group("metadata"+dataset_suffix)
            for dataset_name in metadata_group:
                hdf5_metadata.create_dataset(dataset_name, shape=data_dict[dataset_name].shape, dtype=data_dict[dataset_name].dtype, data=data_dict[dataset_name], compression='gzip', compression_opts=compression_level, chunks=True)

    if (file_path_bin is not None) and (os.path.exists(file_path_bin)):

[docs]def save_to_edf( save_path, filename, data_dict, label_group, data_group, metadata_group, sample_frequency=1250, dataset_suffix=None, path_log='./log.txt'):
    '''Saves raw-data, label-data, and meta-data to separate `.edf` files.

    These files are saved in a standardized way to assure compatibility with other toast packages/modules.

    Important:

        EDF files can only handle up to 16 bit data (-32768 to 32767)

        This can inhibit storing datetime values or timestampes.

    Args:
        save_path (str):        Path where the data should be saved
        filename (str):         Filename prefix to use when generating the data files.
        data_dict (dict):       Dictionary of numpy arrays 
        label_group (list):     Python list of strings (keys in raw_data) identifying the groups for the "_label" file
        data_group (list):      Python list of strings (keys in raw_data) identifying the groups for the "_data" file
        metadata_group (list):  Python list of strings (keys in raw_data) identifying the groups for the "_metadata" file
        sample_frequency (int, optional): Frequency that the data is sampled at.
        dataset_suffix (str, optional):   Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        path_log (str, optional):         Path to the file for outputing logs during execution.
    
    Returns:
        No return value. Generates the "_label" "_data" and "_metadata" files in the save_path locaiton.

    '''
    import pyedflib

    if dataset_suffix is None:
        dataset_suffix = ''
    else:
        dataset_suffix = '-' + dataset_suffix

    filename_label = os.path.join(save_path, filename+'_label'+dataset_suffix+'.edf')
    filename_data  = os.path.join(save_path, filename+'_data'+dataset_suffix+'.edf')
    filename_meta  = os.path.join(save_path, filename+'_metadata'+dataset_suffix+'.edf')

    if len(label_group) > 0:
        tlog.output_log( 'Saving File: '+filename_label, path_log)
        with pyedflib.EdfWriter( filename_label, len(label_group)) as f:
            sample_list = []
            for ch_num, dataset_name in enumerate(label_group):
                sample_list.append( data_dict[dataset_name] )
                f.setLabel(ch_num, dataset_name)
                f.setPhysicalDimension(ch_num, 'boolean')
                f.setSamplefrequency( ch_num, sample_frequency)

            f.writeSamples( sample_list, digital=True)

    if len(data_group) > 0:
        tlog.output_log( 'Saving File: '+filename_data, path_log)
        with pyedflib.EdfWriter( filename_data, len(data_group)) as f:
            sample_list = []
            for ch_num, dataset_name in enumerate(data_group):
                sample_list.append( data_dict[dataset_name] )
                f.setLabel(ch_num, dataset_name)
                f.setPhysicalDimension(ch_num, 'mV')
                f.setSamplefrequency( ch_num, sample_frequency)

            f.writeSamples( sample_list, digital=True)

    if len(metadata_group) > 0:
        tlog.output_log( 'Saving File: '+filename_meta, path_log)
        with pyedflib.EdfWriter( filename_meta, len(metadata_group)) as f:
            sample_list = []
            for ch_num, dataset_name in enumerate(metadata_group):
                sample_list.append( data_dict[dataset_name] )
                f.setLabel(ch_num, dataset_name)
                f.setPhysicalDimension(ch_num, 'misc')
                f.setSamplefrequency( ch_num, sample_frequency)

            f.writeSamples(sample_list, digital=True)
