[docs]def format_and_save( folder_path, filename, save_path, out='wav', source='test', add_datetime=False, label_group=None, data_group=None, metadata_group=None, dataset_suffix=None, rename_firstRepeatedCh=False, path_log='./log.txt'):
    '''High level function that allows data formatting and saving to all happen with one function

    See Also:
        Much of the functionality of this function is determined by other functions. Check :py:func:`~format_data` and :py:func:`~toast.bread.format.save_data.save_to_some_file` for full functionality description.

    Args:
        folder_path (str):               Path to the folder where the data exists
        filename (str):                  The specific filename of the file to process.  Do not include extensions.
        save_path (str):                 File path of where to save the data
        out (str, optional):             Output format to use.  Default is .wav format.
        source (str, optional):          Where the source of the data is. This mechanism helps account for varrying file/folder structures in different data sources.
        add_datetime (bool, optional):   Whether to save the datetime information. Note this adds a lot of processing time and file size
        label_group (list, optional):    Specific labels to save (ordered)
        data_group (list, optional):     Specific data channels to save (ordered)
        metadata_group (list, optional): Specific Metadata channels to save (ordered)
        dataset_suffix (str, optional):  Suffix of filename designating which dataset organization to use. Necessary when following CID convention.
        rename_firstRepeatedCh (bool, optional): If a channel is repeated, whether to rename the FIRST channel to add an index value 0. Indexes 1+ will automatically be added to subsequent channels regardless to avoid name conflicts.
        path_log (str, optional):        Path to the file for outputing logs during execution.

    Returns:  
        True if successful, False if unsuccessful. Generates the "_label" "_data" and "_metadata" files in the save_path location.
        
    '''
    sample_rate = 256
    if label_group is None:
        label_group = ['seizure']
    if data_group is None:
        data_group  = get_eeg_channels( folder_path, filename, source=source, rename_firstRepeatedCh=rename_firstRepeatedCh)
    if metadata_group is None:
        metadata_group = []

    raw_data = format_data( folder_path, filename, source=source, add_datetime=add_datetime, rename_firstRepeatedCh=rename_firstRepeatedCh, path_log=path_log)

    if add_datetime:
        metadata_group.append('datetime_seconds')
        metadata_group.append('datetime_nanosec')

    if raw_data is not None:
        saveData.save_to_some_file( raw_data, label_group, data_group, metadata_group, folder_path, filename, save_path, sample_rate, out=out, source=source, dataset_suffix=dataset_suffix, path_log=path_log)
    else:
        return False

    return True




[docs]def format_data( folder_path, filename, source='test', add_datetime=False, rename_firstRepeatedCh=False, path_log='./log.txt'):
    '''Imports the raw data from the passed filename into a usable dictionary format

    Currently supports the following sources:
    
    * 'test' - Does not alter filepath other than adding extensions. For testing.

    Attention:
        Filenames should be passed without the extension

    Args:
        folder_path (str):             Path to the folder where the data exists
        filename (str):                The specific filename of the file to process.  Do not include extensions.
        source (str, optional):        Where the source of the data is. This mechanism helps account for varrying file/folder structures in different data sources.
        add_datetime (bool, optional): Whether to save the datetime information. Note this adds a lot of processing time and file size
        rename_firstRepeatedCh (bool, optional): If a channel is repeated, whether to rename the FIRST channel to add an index value 0. Indexes 1+ will automatically be added to subsequent channels regardless to avoid name conflicts.
        path_log (str, optional):      Path to the file for outputing logs during execution.

    Returns:
        Dictionary of numpy arrays.  Each element is one channel of data.

    '''
    if source == 'test':
        file_path_edf     = os.path.join( folder_path, filename + ".edf")
        file_path_summary = os.path.join( folder_path, filename.split('_')[0] + "-summary.txt")
        # file_bath_bin   = os.path.join( folder_path, filename + ".bin")

    # Import the raw data.
    data_channels     = {}
    repeated_channels = []
    expected_channels = get_eeg_channels( folder_path, filename, source=source, rename_firstRepeatedCh=rename_firstRepeatedCh)
    tlog.output_log('Reading EDF File', path_log)
    with pyedflib.EdfReader(file_path_edf) as f:
        sample_rate = f.getSampleFrequency(0)
        for indx, signal_header in enumerate( f.getSignalHeaders()):
            if signal_header['label'] not in data_channels.keys():
                data_channels[signal_header['label']] = f.readSignal(indx)
            else:
                tlog.output_log('DUPLICATE CHANNEL '+str(signal_header['label'])+' FOUND IN FILE. ADDING NUMBERING TO REPEATED CHANNEL NAMES TO AVOID NAME CONFLICTS.', path_log)
                count = 1
                new_channel_name = signal_header['label']+'_'+str(count)
                while (new_channel_name in repeated_channels):
                    count += 1
                    new_channel_name = signal_header['label']+'_'+str(count)
                data_channels[new_channel_name] = f.readSignal(indx)
                repeated_channels.append( new_channel_name)
                
    if rename_firstRepeatedCh:
        # For any repeated channels, update the first channel name to include a 0-index suffix
        for ch in repeated_channels:
            if ch[-1] == '1':
                channel_name = ch[:-2]
                data_channels[channel_name+'_0'] = data_channels.pop(channel_name)

    # Fill in any missing channels with zeros
    for channel in expected_channels:
        if channel not in data_channels.keys():
            tlog.output_log('\t\t'+str(channel)+' not in '+str(filename)+'. Replacing with All ZEROS', path_log)
            data_channels[channel] = np.zeros( int(len(data_channels[ list( data_channels.keys())[0]])))

    # Define a time variavle in seconds
    time = np.linspace( 0, len(data_channels[ list(data_channels.keys())[0]])/sample_rate, num=int(len(data_channels[ list( data_channels.keys())[0]])))
    data_channels['time'] = time
    data_channels['seizure'] = np.zeros( len(time))

    tlog.output_log('\tGenerating Annotations', path_log)
    with open( file_path_summary, 'r') as f:
        line = f.readline()
        while filename not in line:
            line = f.readline()
        
        if filename in line:
            start_time = f.readline().strip().split(' ')[-1]
            f.readline() # File end time, not used
            num_seizures = int( f.readline().strip().split(' ')[-1])
            while num_seizures:
                start_index = int( int(  f.readline().strip().split(' ')[-2]) * sample_rate)
                end_index   = int( int( f.readline().strip().split(' ')[-2]) * sample_rate)
                data_channels['seizure'][start_index : end_index] = 1
                num_seizures -=  1
        else:
            tlog.output_log('Filename '+str(filename)+' not found in summary file.  Skipping file.', path_log)
            return None


    # Manipulate Data into expected format.
    # Optimize for saving by reducing file size

    tlog.output_log('\tCasting channels to appropriate types', path_log)
    for channel in expected_channels:
        data_channels[channel] = data_channels[channel].astype(np.int16)
    
    data_channels['seizure']   = data_channels['seizure'].astype(np.int8)
    data_channels['time']      = (data_channels['time']*(10**3)).astype(np.int32)

    # Optionally add a datetime variable, using filename date/time information as starting value
    if add_datetime:
        import datetime
        tlog.output_log('\tCreating datetime values', path_log)

        with open( file_path_summary, 'r') as f:
            last_file_start = False
            # time_elapsed = datetime.timedelta()
            days_elapsed = 0

            line = f.readline()
            while filename not in line:
                line = f.readline()
                if 'File Start Time: ' in line:
                    ts = [int(e) for e in line.strip().split(' ')[3].strip().split(':')]
                    file_start = datetime.datetime( year=1970, month=1, day=1, hour=ts[0], minute=ts[1], second=ts[2])

                    if not last_file_start: # If this is the first file to add.
                        last_file_start = file_start
                    else:
                        if last_file_start.hour > file_start.hour:
                            days_elapsed += 1
                        #     time_elapsed += (file_start + datetime.timedelta(days=1)) - last_file_start
                        # else:
                        #     time_elapsed += file_start - last_file_start

                        last_file_start = file_start

            if filename in line:
                start_time = f.readline().strip().split(' ')[-1]

        # start_time imported above when reading seizure labels
        ts = [int(e) for e in start_time.strip().split(':')]
        if last_file_start and last_file_start.hour > ts[0]:
            days_elapsed += 1
        start_time_datetime = datetime.datetime( year=1970, month=1, day=1+days_elapsed, hour=ts[0], minute=ts[1], second=ts[2])
        datetime_array = start_time_datetime + time * datetime.timedelta(seconds=1)
        # Must convert to numeric timestamp values for storage
        data_channels['datetime_seconds'] = np.array( [int( d.timestamp()) for d in datetime_array]).astype(np.int32)
        data_channels['datetime_nanosec'] = np.array( [int( (d.timestamp()-int(d.timestamp())) * (10**9)) for d in datetime_array]).astype(np.int32)

    return data_channels




[docs]def get_eeg_channels( folder_path, filename, source='test', rename_firstRepeatedCh=False):
    ''' Housekeeping function to get the EEG channel information from the summary.txt file.

    This is used when extracting metadata information (channel ID) about recording files in the CHB-MIT dataset.

    Currently supports the following sources:
    
    * 'test' - Does not alter filepath other than adding extensions. For testing.

    Args:
        folder_path (str):      Path to the folder where the data exists
        filename (str):         The specific filename of the file to process.  Do not include extensions.
        source (str, optional): Where the source of the data is. This mechanism helps account for varrying file/folder structures in different data sources.
        rename_firstRepeatedCh (bool, optional): If a channel is repeated, whether to rename the FIRST channel to add an index value 0. Indexes 1+ will automatically be added to subsequent channels regardless to avoid name conflicts.

    Returns:
        List of extracted channel labels.

    '''
    if source == 'test':
        file_path_summary = os.path.join( folder_path, filename.split('_')[0] + "-summary.txt")
        # file_bath_bin   = os.path.join( folder_path, filename + ".bin")
    
    with open( file_path_summary, 'r') as f:
        found_channelList = False
        channel_list = []
        repeated_channels = []

        for line in f:
            if found_channelList and ('Channel' in line):
                channel_name = line.strip().split(' ')[2]
                if channel_name not in channel_list:
                    channel_list.append( channel_name)
                else:
                    count = 1
                    new_channel_name = channel_name+'_'+str(count)
                    while (new_channel_name in repeated_channels):
                        count += 1
                        new_channel_name = channel_name+'_'+str(count)
                    channel_list.append(      new_channel_name)
                    repeated_channels.append( new_channel_name)

            else:
                found_channelList = False
            
            if 'Channels' in line: # Keyword for header to channels section
                found_channelList = True
                f.readline() # Skip line of asterisks
                continue
        
        if rename_firstRepeatedCh:
            for ch in repeated_channels:
                if ch[-1] == '1':
                    channel_name = '_'.join( ch.split('_')[:-1])
                    i_replace = channel_list.index(channel_name)
                    channel_list[i_replace] = channel_name+'_0'
    
    return channel_list
