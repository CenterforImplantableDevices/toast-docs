[docs]def extract_segments( all_labels_dataframe, sample_rate_expected, sample_length, num_segments, path_log='./log.txt'):
    '''High level function that helps use labels assigned by label_data to extract metadata that identifies segments within an event.

    Args: 
        all_labels_dataframe (dataframe): Dataframe containing all raw data and labels
        sample_rate_expected (int):       Sample rate the data was recorded at (Hz)
        sample_length (int):              How many data points in a sample. Commonly calculated as seconds*fs
        num_segments (list):              List of the number of segments to extract for each label. There are currently three options for number of segments requested:

                                * num_segments >  0 : extract num_segments number of segments
                                * num_segments = -1 : extract the maximum number of non-overlapping segments, with the number of segments equal for each label.
                                * num_segments = -2 : extract the maximum number of non-overlapping segments as possible for each label.
                                * EX: [0, 0, 50, 25] would extract 0 segments for label 0 and 1, 50 segments for label 2, and 25 segments for label 3
        
        path_log (str, optional):         Path to the file for outputing logs during execution.

    Returns:
        List of dictionaries identifying the metadata required to identify the requested segments.
        See CID convention for this dictionary in toast/bread/README.md
    '''
    # Returns a list of a list of dictionaries describing segments
    # Outer list designates the label of the enclosed segments
    # The inner list is a collection of dictonaries, with each dictionary representing a sample
    # Each dictionary contains feilds describing the ssoicated filenames, start_index, stop_index, start_datetime, stop_datetime, and label_int
    data_segments = []
    fewest_possible_segments_real = False
    for label_int in range( len( num_segments)):
        data_segments.append([])
        if label_int not in all_labels_dataframe['label_int'].values:
            tlog.output_log('No segments found for label '+str(label_int), path_log)
            continue
        data_to_sample = all_labels_dataframe[ all_labels_dataframe['label_int'] == label_int].copy()
        data_to_sample.reset_index(inplace=True, drop=True)
        data_to_sample['sample_history'] = 0
        if num_segments[label_int] > 0:
            tlog.output_log('Extracting '+str(num_segments[label_int])+' segments for label '+str(label_int), path_log)
            data_segments[label_int] = extract_segments_for_label( data_to_sample, sample_rate_expected, sample_length, num_segments[label_int], path_log=path_log)
            print()

        elif num_segments[label_int] == -1:
            # Find the maximum sample count possible for all output types
            fewest_possible_segments_est = sys.maxsize
            for i, num in enumerate( num_segments):
                if num == -1:
                    possible_segments = int(len( all_labels_dataframe[ all_labels_dataframe['label_int'] == i]) / sample_length)
                    if possible_segments < fewest_possible_segments_est:
                        fewest_possible_segments_est = possible_segments
                        fewest_possible_index = i
            if label_int == fewest_possible_index:
                tlog.output_log('Extracting ~'+str(fewest_possible_segments_est)+' segments for label '+str(label_int)+' (max)', path_log)
                data_segments[label_int] = extract_max_segments_for_label( data_to_sample, sample_rate_expected, sample_length)
                fewest_possible_segments_real = len( data_segments[label_int])
            else:
                tlog.output_log('Extracting ~'+str(fewest_possible_segments_est)+' segments for label '+str(label_int), path_log)
                data_segments[label_int] = extract_segments_for_label( data_to_sample, sample_rate_expected, sample_length, fewest_possible_segments_est, path_log=path_log)
                if fewest_possible_segments_real:
                    data_segments = [sample[:fewest_possible_segments_real] if len(sample)>0 else [] for sample in data_segments ]
                print()

        elif num_segments[label_int] == -2:
            tlog.output_log('Extracting'+str(int(len( all_labels_dataframe[ all_labels_dataframe['label_int'] == label_int])/sample_length))+' segments for label '+str(label_int), path_log)
            data_segments[label_int] = extract_max_segments_for_label( data_to_sample, sample_rate_expected, sample_length)


    return data_segments





[docs]def extract_segments_for_label( data_to_sample, sample_rate_expected, sample_length, num_segments, path_log='./log.txt'):
    '''Internal function used to sample and extract metadata meeded to identify each sample.
    Used when an explicit number of samples is needed.

    Args:   
        data_to_sample (dataframe): Dataframe containing all raw data and labels
        sample_rate_expected (int): Sample rate the data was recorded at (Hz)
        sample_length (int):        How many data points in a sample. Commonly calculated as seconds*fs
        num_segments (int):         Number of samples to extract.
        path_log (str, optional):   Path to the file for outputing logs during execution.
            
    Returns:
        List of dictionaries identifying the metadata required to identify the requested samples.
        See CID convention for this dictionary in toast/bread/README.md
    '''
    data_segments = []
    data_to_sample['sample_history'] = 0
    i_filename          = data_to_sample.columns.get_loc('filename')
    i_label_int         = data_to_sample.columns.get_loc('label_int')
    i_file_frame        = data_to_sample.columns.get_loc('file_frame')
    i_datetime_ts       = data_to_sample.columns.get_loc('datetime_ts')
    i_samp_hist         = data_to_sample.columns.get_loc('sample_history')
    i_time_beforeEvent  = data_to_sample.columns.get_loc('time_beforeEvent')
    i_time_afterEvent   = data_to_sample.columns.get_loc('time_afterEvent')
    # i_time_duringEvent  = data_to_sample.columns.get_loc('time_duringEvent')

    total_data_length = len(data_to_sample)
    segments_added    = 0
    count_resample    = 0
    last_new_segments = 0
    while len(data_segments) < num_segments:
        num_new_segments = num_segments - len(data_segments)

        segments = sample( range( len(data_to_sample)), k=num_new_segments)
        for s in segments:
          if (data_to_sample.iloc[s:s+sample_length, i_samp_hist].sum() == 0):                                        # If the entire sample hasn't already been sampled
            if ((data_to_sample.iloc[s:s+sample_length, i_datetime_ts].diff().abs() > 2/sample_rate_expected).sum() == 0):  # If the segments are continuous
              if (s+sample_length < total_data_length):                                                               # If the sample length doesn't extend past the end of the data
                data_segments.append( {  'filename':           data_to_sample.iat[s, i_filename],
                                        'label_int':          data_to_sample.iat[s, i_label_int],
                                        'start_index':        data_to_sample.iat[s, i_file_frame],
                                        'stop_index':         data_to_sample.iat[s+sample_length, i_file_frame],
                                        'start_datetime':     datetime.fromtimestamp( data_to_sample.iat[s, i_datetime_ts]),
                                        'stop_datetime':      datetime.fromtimestamp( data_to_sample.iat[s+sample_length, i_datetime_ts]),
                                        'time_beforeEvent':   data_to_sample.iat[s, i_time_beforeEvent],
                                        'time_afterEvent':    data_to_sample.iat[s, i_time_afterEvent]
                                        # 'time_duringEvent':   data_to_sample.iat[s, i_time_duringEvent]
                                      })
                data_to_sample.iloc[s:s+sample_length, i_samp_hist] = 1
                segments_added += 1
                progress.progress_bar(segments_added, num_segments)
        
        if num_new_segments == last_new_segments:
            count_resample += 1
            if count_resample > 500:
                tlog.output_log('COULD NOT FIND ENOUGH NON-OVERLAPPING SEGMENTS.  RETURNING '+str(len(data_segments))+' SEGMENTS INSTEAD OF '+str(num_segments), path_log)
                return data_segments
        else:
            count_resample = 0
            last_new_segments = num_new_segments
    
    return data_segments





[docs]def extract_max_segments_for_label( data_to_sample, sample_rate_expected, sample_length):
    '''Internal function used to sample and extract metadata meeded to identify each sample.
    Used when the maximum available number of samples is desired.

    Args:
        data_to_sample (dataframe):  Dataframe containing all raw data and labels
        sample_rate_expected (int):  The sample rate the data was recorded at (Hz)
        sample_length (int):         How many data points in a sample. Commonly calculated as seconds*fs
            
    Returns:
        List of dictionaries identifying the metadata required to identify the requested samples.
        See CID convention for this dictionary in toast/bread/README.md
    '''
    data_segments = []
    i_filename          = data_to_sample.columns.get_loc('filename')
    i_label_int         = data_to_sample.columns.get_loc('label_int')
    i_file_frame        = data_to_sample.columns.get_loc('file_frame')
    i_datetime_ts       = data_to_sample.columns.get_loc('datetime_ts')
    i_time_beforeEvent  = data_to_sample.columns.get_loc('time_beforeEvent')
    i_time_afterEvent   = data_to_sample.columns.get_loc('time_afterEvent')
    # i_time_duringEvent  = data_to_sample.columns.get_loc('time_duringEvent')

    data_to_sample['continuous_data'] = data_to_sample['datetime_ts'].diff().abs() < 2/sample_rate_expected
    data_to_sample.iloc[-1, data_to_sample.columns.get_loc('continuous_data')] = False # Set the very last value to false to allow loop to finish iterating
    last_continuous_index = 0
    for continuous_index, _ in data_to_sample[ data_to_sample['continuous_data'] == False ].iterrows():
        session_data = data_to_sample.iloc[ last_continuous_index:continuous_index, :].copy()
        session_data.reset_index( inplace=True, drop=True)
        start_offset = int( ((len(session_data)-sample_length) % sample_length) / 2)
        for s in range( start_offset, len(session_data)-sample_length, sample_length):
            data_segments.append( { 'filename':           session_data.iat[s, i_filename],
                                    'label_int':          session_data.iat[s, i_label_int],
                                    'start_index':        session_data.iat[s, i_file_frame],
                                    'stop_index':         session_data.iat[s+sample_length, i_file_frame],
                                    'start_datetime':     datetime.fromtimestamp( session_data.iat[s, i_datetime_ts]),
                                    'stop_datetime':      datetime.fromtimestamp( session_data.iat[s+sample_length, i_datetime_ts]),
                                    'time_beforeEvent':   session_data.iat[s, i_time_beforeEvent],
                                    'time_afterEvent':    session_data.iat[s, i_time_afterEvent]
                                    # 'time_duringEvent':   data_to_sample.iat[s, i_time_duringEvent]
                                  })

        last_continuous_index = continuous_index

    shuffle( data_segments)
    return data_segments





[docs]def extract_labelRanges( all_labels_dataframe, sample_rate_expected, labels_to_segment):
    '''Use labels assigned by label_data to extract metadata that identifies overall data chunks enclosing an event.

    Example of labels_to_segment: If labels_to_segment = [[3], [2,1]], then ranges containing label_int 3 will be identified, 
    and ranges containg either label_int 2 or label_int 1 will be gropued together and identified

    Args:
        all_labels_dataframe (dataframe): Dataframe containing all raw data and labels
        sample_rate_expected (int):       The rate which data is sampled at (Hz).
        labels_to_segment (list):         List of lists identifying which labels to segment together. Example above.
            
    Returns:
        List of dictionaries identifying the metadata required to identify the requested ranges.
        See CID convention for this dictionary in toast/bread/README.md
    '''
    data_ranges = []

    for segment_index, segment in enumerate( labels_to_segment):
        data_ranges.append([])
        data_to_sample = all_labels_dataframe[ all_labels_dataframe['label_int'].isin(segment)].copy()
        data_to_sample.reset_index(inplace=True, drop=True)
        
        i_filename         = data_to_sample.columns.get_loc('filename')
        i_label_int        = data_to_sample.columns.get_loc('label_int')
        i_file_frame       = data_to_sample.columns.get_loc('file_frame')
        i_datetime_ts      = data_to_sample.columns.get_loc('datetime_ts')
        i_time_beforeEvent = data_to_sample.columns.get_loc('time_beforeEvent')
        i_time_afterEvent  = data_to_sample.columns.get_loc('time_afterEvent')
        i_time_duringEvent = data_to_sample.columns.get_loc('time_duringEvent')
        last_row_index = 0
        for row_index, _ in data_to_sample[ data_to_sample[ 'datetime_ts'].diff().abs() > 2/sample_rate_expected].iterrows():
            data_ranges[segment_index].append( { 'filename':            data_to_sample.iat[row_index-1, i_filename],
                                                   'label_int':         data_to_sample.iat[row_index-1, i_label_int],
                                                   'start_index':       data_to_sample.iat[last_row_index, i_file_frame],
                                                   'stop_index':        data_to_sample.iat[row_index-1, i_file_frame],
                                                   'start_datetime':    datetime.fromtimestamp( data_to_sample.iat[last_row_index, i_datetime_ts]),
                                                   'stop_datetime':     datetime.fromtimestamp( data_to_sample.iat[row_index-1, i_datetime_ts]),
                                                   'time_beforeEvent':     data_to_sample.iat[row_index-1, i_time_beforeEvent],
                                                   'time_afterEvent':  data_to_sample.iat[row_index-1, i_time_afterEvent],
                                                   'time_duringEvent':    data_to_sample.iat[row_index-1, i_time_duringEvent]
                                                  })
            last_row_index = row_index

        # Add the last data segment
        row_index = len(data_to_sample)-1
        if len( data_to_sample) > 0:
            data_ranges[segment_index].append( { 'filename':            data_to_sample.iat[row_index, i_filename],
                                                   'label_int':         data_to_sample.iat[row_index, i_label_int],
                                                   'start_index':       data_to_sample.iat[last_row_index, i_file_frame],
                                                   'stop_index':        data_to_sample.iat[row_index, i_file_frame],
                                                   'start_datetime':    datetime.fromtimestamp( data_to_sample.iat[last_row_index, i_datetime_ts]),
                                                   'stop_datetime':     datetime.fromtimestamp( data_to_sample.iat[row_index, i_datetime_ts]),
                                                   'time_beforeEvent':     data_to_sample.iat[row_index, i_time_beforeEvent],
                                                   'time_afterEvent':  data_to_sample.iat[row_index, i_time_afterEvent],
                                                   'time_duringEvent':    data_to_sample.iat[row_index, i_time_duringEvent]
                                                 })
    
    return data_ranges
