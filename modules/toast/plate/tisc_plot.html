[docs]def make_square( input_data, data_dim=0, scale_multiplier=2, normalize=True):
    '''Separates TiSc data into a 2D array with each scale in one row, with each scale stretched to make an even length 2D array.

    In this output each column index correlates wtih a unit of time.

    Args:
        input_data (numpy array):         TiSc organized data
        data_dim (int, optional):         Dimension along which the TiSc organized data is organized.
        scale_multiplier (int, optional): Exponential base designating how fast the window size increases.
        normalize (bool, optional):       Whether to independently normalize each scale to between 0 and 1.

    Returns:
        Numpy ndarray: 2D arra where rows are each scale the the columns are each time index.
    '''
    # Get the starting index for the smallest scale (smallest window, most values), and the number of levels to iterate for
    start_index = int( (input_data.shape[data_dim]+1) / scale_multiplier)
    num_levels  = int( math.log( input_data.shape[data_dim], scale_multiplier))+1

    # Create the output array
    output = torch.zeros((num_levels, start_index))
    num_repeats = 1
    # Iterate thorugh each scale and insert the associated scale values into the proper index in the output array.
    for i in range( num_levels):
        # Expand the length of the data by using the repeat_interleave function
        output[i, :] = input_data[ start_index-1 : start_index*scale_multiplier-1].repeat_interleave( num_repeats)
        # If we normalize each scale, divide by num_repeats
        # wich increases as the scale gets larger (i.e. window size and output values get bigger)
        if normalize:
            output[i, :] = output[i, :].div(num_repeats)

        # Increment the indexing values
        start_index  = int( start_index / scale_multiplier)
        num_repeats *= scale_multiplier
    
    return output

    '''Separates TiSc data into a 2D array with each scale in one row, where each scale remains its original length.

    In this output, each row will exponentailly decrease in length. The remaining values which have no meaning are set to zero.

    Args:
        input_data (numpy array):         TiSc organized data
        data_dim (int, optional):         Dimension along which the TiSc organized data is organized.
        scale_multiplier (int, optional): Exponential base designating how fast the window size increases.
        normalize (bool, optional):       Whether to independently normalize each scale to between 0 and 1.

    Returns:
        Numpy ndarray: 2D arra where rows are each scale the the columns the scale index.
    '''
    # Get the starting index for the smallest scale (smallest window, most values), and the number of levels to iterate for
    start_index = int( (input_data.shape[data_dim]+1) / scale_multiplier)
    num_levels  = int( math.log( input_data.shape[data_dim], scale_multiplier))+1

    # Create the output array
    output = torch.zeros((num_levels, start_index))
    num_repeats = 1
    # Iterate thorugh each scale and insert the associated scale values into the proper index in the output array.
    for i in range( num_levels-1, -1, -1):
        # Expand the length of the data by using the torch_pad function to padd with zeros.
        output[i, :] = torch_pad( input_data[ start_index-1 : start_index*scale_multiplier-1], (0, output.shape[1]-start_index), mode='constant', value=0)
        # If we normalize each scale, divide by num_repeats
        # wich increases as the scale gets larger (i.e. window size and output values get bigger)
        if normalize:
            output[i, :] = output[i, :].div(num_repeats)

        # Increment the indexing values
        start_index  = int( start_index / scale_multiplier)
        num_repeats *= scale_multiplier
    
    return output




[docs]def make_triangle_list( input_data, data_dim=0, scale_multiplier=2, normalize=True):
    '''Separates TiSc data into a 1D python list of 1D Numpy arrays were each scale is a list element. Each scale remains its original length.

    Args:
        input_data (numpy array):         TiSc organized data
        data_dim (int, optional):         Dimension along which the TiSc organized data is organized.
        scale_multiplier (int, optional): Exponential base designating how fast the window size increases.
        normalize (bool, optional):       Whether to independently normalize each scale to between 0 and 1.

    Returns:
        List where each element is the scale, containing a 1D numpy array where the columns are the scale index.
    '''
    # Get the starting index for the smallest scale (smallest window, most values), and the number of levels to iterate for
    start_index = int( (input_data.shape[data_dim]+1) / scale_multiplier)
    num_levels  = int( math.log( input_data.shape[data_dim], scale_multiplier))+1

    # Create the output array
    output = []
    num_repeats = 1
    # Iterate thorugh each scale and append the associated scale values to the proper index in the output array.
    for i in range( num_levels-1, -1, -1):
        out = input_data[ start_index-1 : start_index*scale_multiplier-1]
        # If we normalize each scale, divide by num_repeats
        # wich increases as the scale gets larger (i.e. window size and output values get bigger)
        if normalize:
            out = out.div(num_repeats)
        
        output.append(out.detach().unsqueeze(0).numpy())

        # Increment the indexing values
        start_index  = int( start_index / scale_multiplier)
        num_repeats *= scale_multiplier
    
    # Since we built the output array backwards, reverse it before returning
    return output[::-1]

    '''Uses a network object to separate the weights of the first tisc layer, sort into the proper channels, and plot the weights of the channel.

    Function assumes that weights of each channel alternate within each filter window, matching the typical .wav file organization.

    Function assumes the network as arranges following convention of nested sequential objets, as defined below:
    
        * Sequential[0]       contains all TiSc-based network processing.
        * Sequential[0][*]    contains one or many TiSc channels
        * Sequential[0][*][0] contains a :py:class:`~toast.heat.timescaleL.TSC_input` layer to initially process the input data

    Args:
        network (toast network):              A toast.heat generated network that has been trained so that it contains training history values to plot.
        num_channels_data (int):              Number of data channels to separate out.
        description (str, optional):          A description of the data in the plot. This will be used as the filename.
        show_plot (bool, optional):           Whether to show the plot after generation
        plot_dpi (int, optional):             Resolution of the generated/saved plots.
        plot_linewidth (float, optional):     Linewidth to use in the generated/saved plots.
        save_plot (bool, optional):           Whether to save the plot after generation.  Note the filename will be determined based on attributes of the network's path_network attribute
        skip_plot (bool, optional):           Whether to skip the plotting funcions. Use if only want to save data in pickle file.
        save_data (bool, optional):           Whether to save the weight data in a pickle file after organization.  Note the filename will be determined based on attributes of the network's path_network attribute
        save_path (str, optional):            Path to the parent directory where output plots should be saved.
        output_folder_name (str, optional):   Child directory to save the plots to.
        path_log (str, optional):             Path to the log file for logging activity.
        bkg_color (str -or- tuple, optional): Character or RBG Thruple representing the background color of the plot.
        close_plot (bool, optional):            Whether to close the figure when finished. THis helps minimize memory usage.
        use_agg (bool, optional):             Whether to use the Agg backend of matplotlib.  Use when connecting to non-display machines/servers

    Returns:
        No return value. Generates and displays/saves a plot.
        If plots are saved, a pickle file with the raw weight values will also be saved. File contians one list object with dimensions data[channel_tisc][scale][channel_eeg][raw_weights]
    '''
    if use_agg:
        import matplotlib
        matplotlib.use('Agg')
    import matplotlib.pyplot as plt

    # If no description exists, create one
    if len(description) == 0:
        description = network.path_network.split('/')[-1]
    print('Plotting ', description, end='\r')

    # Extract the assoiciated layer parameter values.  These are the raw weights that we will plot.
    tisc_weights_channels = []
    for name, param in network.named_parameters():
        # If in tisc nn.Sequential grp and Is the first element of nn.Sequential for that tisc channel
        if (name.split('.')[1] == '0') and (name.split('.')[3] == '0') and (name.split('.')[4].lower().startswith('weight')):
            # don't forget to move the data to the cpu to speed up operations.
            tisc_weights_channels.append( param.data.to( torch.device('cpu')))
        elif (name.split('.')[0] == 'layers_encode') and (name.split('.')[2] == '0') and (name.split('.')[3].lower().startswith('weight')):
            # don't forget to move the data to the cpu to speed up operations.
            tisc_weights_channels.append( param.data.to( torch.device('cpu')))

    # Plot each tisc channel indepedently
    weights_all = []
    for channel_tisc, tisc_weights in enumerate( tisc_weights_channels):
        # Separate the extracted weights into the respective scales.
        scale_weights = make_triangle_list( tisc_weights)

        # Analyze each scale independently
        weights_scale_data = []
        for scale in scale_weights:
            # For each scale separate each eeg channel by extracting only the num_channels'th value.
            channels_data = []
            for i in range( num_channels_data):
                channels_data.append( scale[0][i::num_channels_data])
            
            # Add the separated channel values to the weights_scale_data for that scale
            weights_scale_data.append(channels_data)
        
        if save_data:
            weights_all.append(weights_scale_data)

        if not skip_plot:
            # Lots of figures are about to open, so lets silent that warning
            plt.rcParams.update({'figure.max_open_warning': 0})
            # Iterate through each scale
            for scale in range(len(weights_scale_data)):
                filename = description+'_wgts-tisc-'+str(channel_tisc)+'-sca-'+str(scale)+'-len-'+str(len(weights_scale_data[scale][0]))

                # plt.figure(figsize=(16,9)) #, dpi=plot_dpi)
                _, ax0 = plt.subplots()
                ax0.set_facecolor(bkg_color)
                ax0.set_title('TISC '+str(channel_tisc)+' : SCALE '+str(scale)+' : Length '+str(len(weights_scale_data[scale][0]))+'\n'+description)
                # Plot each channel in its own
                for ch in range(num_channels_data):
                    plt.subplot(num_channels_data, 1, ch+1)
                    plt.plot( weights_scale_data[scale][ch], linewidth=plot_linewidth, color='k')
                    plt.yticks([],[])

                plt.subplots_adjust(wspace=0, hspace=0)

                if save_plot:
                    if not os.path.exists(os.path.join( save_path, output_folder_name)):
                        os.mkdir(os.path.join( save_path, output_folder_name))
                    tlog.output_log('Saving : '+str(os.path.join( save_path, output_folder_name, filename+'.png')), path_log)
                    plt.savefig(os.path.join( save_path, output_folder_name, filename+'.png'), dpi=plot_dpi)
                if close_plot:
                    plt.close()

    if save_data:
        # Pickle file will contain one object, a list with dimensions [tisc_channel][scale][eeg_channel][raw_weights]
        with open( os.path.join( save_path, output_folder_name, description+'.pkl'), 'wb') as f:
            pickle.dump( weights_all, f)

    if show_plot:
        plt.show()

        No return value. Generates a plot for the user to either be saved here or viewed later by the user.
    '''
    if use_agg:
        import matplotlib
        matplotlib.use('Agg')
    import matplotlib.pyplot as plt

    if not contains_kfolds:
        gradCAM_expanded = [gradCAM_expanded]
    if description == '':
        description = 'gradCAM-summary'
    print('Plotting ', description, end='\r')

    for output_class in range( len( gradCAM_expanded[0])):
        # Omit the time dimension, as we will sum over that as we append the data
        gradCAM_expanded_all = gradCAM_expanded[0][output_class] # .sum(dim=-1)
        if len(gradCAM_expanded) > 1:
            for kfold in gradCAM_expanded[1:]:
                # Sum accross the time dimension
                gradCAM_expanded_all = torch.cat((gradCAM_expanded_all, kfold[output_class]), dim=1) # .sum(dim=-1)), dim=1)

        # Normalize column-wise
        gradCAM_expanded_all = gradCAM_expanded_all.divide( gradCAM_expanded_all.max(dim=2,keepdim=True)[0])

        numTotal_layers = gradCAM_expanded_all.shape[0]
        plt.figure(figsize=(9,16))
        for index_layer, data_layer in enumerate( gradCAM_expanded_all):
            ax0 = plt.subplot(numTotal_layers,1,index_layer+1)
            plt.imshow( data_layer.transpose(0,1), aspect='auto')

            else:
                plt.ylabel('TS'+str(index_layer))
            if colorbar:
                plt.colorbar()
            if index_layer == 0:
                plt.title(title+'\nClass '+str(output_class)+' gradCAM Weighted Activation Mag.')
            if index_layer == numTotal_layers:
                plt.xlabel(x_label)
            else:
                ax0.set_xticklabels([])

        plt.tight_layout()
        
        if save_txt:
            output = []
            output.append('\n\nOUTPUT CLASS'+str(output_class)+' ACTIVATIONS:\n')
            for scale in range(gradCAM_expanded_all.shape[2]):
                output.append('SCALE '+str(scale))
                scale_data = gradCAM_expanded_all[:,:,scale]
                output.append('\tmean:  {:.6e}\tstdev:  {:.6e}\t, min:  {:.6e}\t, max:  {:.6e}\t'.format(scale_data.mean().item(), scale_data.std().item(), scale_data.min().item(), scale_data.max().item()))

            if not os.path.exists(os.path.join( save_path, output_folder_name)):
                os.mkdir(os.path.join( save_path, output_folder_name))

            with open( os.path.join( save_path, output_folder_name, description+'.txt'), 'a') as f:
                for line in output:
                    f.write(line)

        if show_plot:
            plt.show()
        if save_plot:
            if not os.path.exists(os.path.join( save_path, output_folder_name)):
                os.mkdir(os.path.join( save_path, output_folder_name))
            tlog.output_log('Saving : '+str(os.path.join( save_path, output_folder_name, description+'_outputClass-'+str(output_class)+'.png')), path_log)
            plt.savefig(os.path.join( save_path, output_folder_name, description+'_outputClass-'+str(output_class)+'.png'), dpi=plot_dpi)
        if close_plot:
            plt.close()
